**Step 2 — Insert `pullChunkPhase6` AFTER `defaultPullChunkPhase2` (around line 210):**

```typescript
// ─── Phase 6 pullChunk (real adapter invocation) ──────────────────────────
//
// Phase 2 deferred real scraper invocation to Phase 6 (per
// .planning/phases/02-.../02-VERIFICATION.md::deferred). Wave 4 wires this:
// each chunk calls getAdapter(platform, {tenantId}).fetch{Orders,Shifts,...}
// and upserts the normalized rows. Per-capability failures are caught + the
// chunk continues; the whole-chunk failure is recorded as a FAILED IngestRun
// row and the function returns {rowsOk: 0} (Pitfall 6 — never throw to abort
// the 30-day window).

export async function pullChunkPhase6(args: BackwashChunkArgs): Promise<{ rowsOk: number }> {
  const { tenantId, platform, from, to } = args;
  const range: DateRange = {
    from: new Date(`${from}T00:00:00Z`),
    to: new Date(`${to}T00:00:00Z`),
  };

  const startedAt = new Date();
  let rowsIn = 0;
  let rowsOk = 0;
  const errors: string[] = [];

  let adapter;
  try {
    adapter = getAdapter(platform as Platform, { tenantId });
  } catch (err: any) {
    await writeIngestRun({
      tenantId, platform, source: "BACKWASH", status: "FAILED",
      startedAt, finishedAt: new Date(), rowsIn: 0, rowsOk: 0,
      errorLog: `getAdapter failed: ${err?.message ?? String(err)}`,
    });
    return { rowsOk: 0 };
  }

  // Each capability runs in its own try/catch — a failing scraper doesn't
  // kill the XLSX/mobile fetches.
  const safeFetch = async (
    label: string,
    fn?: (tid: string, r: DateRange) => Promise<NormalizedRow<unknown>[]>,
  ): Promise<NormalizedRow<unknown>[]> => {
    if (!fn) return [];
    try {
      const rows = await fn.call(adapter, tenantId, range);
      rowsIn += rows.length;
      return rows;
    } catch (err: any) {
      if (err instanceof NotAvailable) {
        // Expected — composite already fell through; nothing to record.
        return [];
      }
      errors.push(`${label}: ${err?.message ?? String(err)}`);
      return [];
    }
  };

  const orderRows = await safeFetch("fetchOrders", adapter.fetchOrders);
  const shiftRows = await safeFetch("fetchShifts", adapter.fetchShifts);
  const attendanceRows = await safeFetch("fetchAttendance", adapter.fetchAttendance);
  // Cash is XLSX-import-only (per phase contract). Backwash NEVER produces
  // cash rows — adapter.fetchCash returns [] across all 4 platforms. We do
  // NOT call it from the backwash worker; cash arrives via /api/.../import
  // XLSX uploads. cashRows is hardcoded to 0 to keep audit-row math honest.
  const cashRows = 0;
  const violationRows = await safeFetch("fetchViolations", adapter.fetchViolations);

  // v1 upsert path: OrderLog only. Shifts/attendance/cash/violations are
  // counted but not yet upserted (Phase 11 expands the upsert mappings).
  // Wave 4 keeps Phase 6's blast radius minimal — Pitfall 1 (don't bloat
  // adapters or workers during a refactor wave).
  for (const row of orderRows) {
    try {
      const data = row.data as any;
      if (!data?.id || !data?.platformOrderId) continue;
      // Upsert against existing OrderLog @@unique (verify exact key in
      // schema.prisma; the executor reads schema and matches).
      // Phase 5's mobile fetcher already produces OrderLog rows — for those,
      // the upsert is a no-op (same row already present); for scraper rows,
      // it inserts.
      await prisma.orderLog.upsert({
        where: { id: data.id },
        create: { ...data, tenantId },
        update: { ...data },
      });
      rowsOk++;
    } catch (err: any) {
      errors.push(`OrderLog upsert: ${err?.message ?? String(err)}`);
    }
  }
  // Shift/attendance/cash/violation rows: count toward rowsOk for audit
  // visibility but do NOT upsert in Phase 6 (Phase 11 wires per-capability
  // mappings once mobile + scraper produce their full normalized shapes).
  // Note: cashRows excluded (always 0; XLSX-import-only contract).
  rowsOk += shiftRows.length + attendanceRows.length + violationRows.length;

  const status =
    errors.length === 0 ? "SUCCESS" :
    rowsOk > 0 ? "PARTIAL" :
    "FAILED";

  await writeIngestRun({
    tenantId, platform, source: "BACKWASH", status,
    startedAt, finishedAt: new Date(),
    rowsIn, rowsOk,
    errorLog: errors.length === 0 ? null : errors.join("\n"),
  });

  return { rowsOk };
}
```
