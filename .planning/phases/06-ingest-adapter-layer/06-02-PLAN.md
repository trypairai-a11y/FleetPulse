**Step 1 — `backend/src/services/ingest/keeta/scraper.ts` (~70 lines):**

```typescript
// Phase 6 Wave 2 — KeetaScraperAdapter.
// Refactored from queues/keetaPortalScraperWorker.ts. The queues file's
// scheduler (setInterval, startKeetaPortalScraperScheduler) stays where it
// is in Wave 2 — Wave 4 may rewire it to import from this adapter.
//
// Until Phase 11 wires real Playwright (REQ-ingest-partner-api-conversations),
// fetchOrders returns [] for tenants WITH creds (preserving the existing
// PARTIAL/no-rows scaffold behavior) and throws NotAvailable for tenants
// without creds.

import { prisma } from "../../../config";
import { decryptCred, hasEncryptedShape } from "../../../utils/portalCreds";
import { logger } from "../../../config/logger";
import {
  AdapterSource, DateRange, IngestAdapter, NormalizedRow, NotAvailable, Platform,
} from "../types";

interface PortalCreds { username: string; password: string }

export class KeetaScraperAdapter implements IngestAdapter {
  readonly platform: Platform = "KEETA";
  readonly source: AdapterSource = "PORTAL_SCRAPER";

  async isAvailable(tenantId: string): Promise<boolean> {
    const settings = await prisma.platformSettings.findUnique({
      where: { tenantId_platform: { tenantId, platform: "KEETA" } },
    });
    const pc = (settings?.notificationConfig as any)?.portalCredentials;
    return Boolean(pc?.username && hasEncryptedShape(pc.password));
  }

  private async loadCreds(tenantId: string): Promise<PortalCreds | null> {
    const settings = await prisma.platformSettings.findUnique({
      where: { tenantId_platform: { tenantId, platform: "KEETA" } },
    });
    const pc = (settings?.notificationConfig as any)?.portalCredentials;
    if (!pc?.username || !hasEncryptedShape(pc.password)) return null;
    return { username: pc.username, password: decryptCred(pc.password) };
  }

  async fetchOrders(tenantId: string, _range: DateRange): Promise<NormalizedRow<unknown>[]> {
    const creds = await this.loadCreds(tenantId);
    if (!creds) throw new NotAvailable("No Keeta portal credentials configured for tenant");
    // TODO Phase 11: real Playwright scrape (login → navigate → extract).
    // Until then, return [] (matches queues/keetaPortalScraperWorker scaffold semantics).
    logger?.info?.({ tenantId }, "[KeetaScraperAdapter] scaffold fetchOrders (no-op)");
    return [];
  }

  async fetchShifts(tenantId: string, _range: DateRange): Promise<NormalizedRow<unknown>[]> {
    const creds = await this.loadCreds(tenantId);
    if (!creds) throw new NotAvailable("No Keeta portal credentials configured for tenant");
    return [];
  }
```

```typescript
// Phase 6 Wave 2 — Talabat XLSX MVP schema (orchestrator resolution #2).
// Founder will refine when sample exports arrive (Phase 11 polish).
// Pitfall 10: header validation BEFORE any row parsing.

import { z } from "zod";
import { parseLocalDate } from "../normalize";

export const REQUIRED_TALABAT_COLUMNS = [
  "date",
  "driver_id",
  "orders_count",
  "online_minutes",
  "attendance_status",
] as const;

export type TalabatXlsxRow = {
  date: Date;
  driverId: string;
  ordersCount: number;
  onlineMinutes: number;
  attendanceStatus: string;
};

export function validateTalabatXlsxHeaders(headerRow: string[]): void {
  const lowercased = headerRow.map((h) => String(h ?? "").trim().toLowerCase());
  const missing = REQUIRED_TALABAT_COLUMNS.filter((req) => !lowercased.includes(req));
  if (missing.length > 0) {
    throw new Error(`Talabat XLSX missing required columns: ${missing.join(", ")}`);
  }
}

const RowSchema = z.object({
  date: z.union([z.string(), z.number(), z.date()]).transform((v) => parseLocalDate(v)),
  driver_id: z.union([z.string(), z.number()]).transform((v) => String(v).trim()),
  orders_count: z.coerce.number().int().nonnegative(),
  online_minutes: z.coerce.number().int().nonnegative(),
  attendance_status: z.string().transform((s) => s.trim().toUpperCase()),
});

export function parseTalabatRow(headers: string[], rowArr: any[]): TalabatXlsxRow {
  const idx = (col: string) => headers.findIndex((h) => h.toLowerCase().trim() === col);
  const obj = {
    date: rowArr[idx("date")],
    driver_id: rowArr[idx("driver_id")],
    orders_count: rowArr[idx("orders_count")],
    online_minutes: rowArr[idx("online_minutes")],
    attendance_status: rowArr[idx("attendance_status")],
  };
  const parsed = RowSchema.parse(obj);
  return {
    date: parsed.date,
    driverId: parsed.driver_id,
    ordersCount: parsed.orders_count,
    onlineMinutes: parsed.online_minutes,
    attendanceStatus: parsed.attendance_status,
  };
}
```

```typescript
// Phase 6 Wave 2 — TalabatXlsxAdapter.
// Pitfall 9: idempotent upsert on TalabatDailyMetrics compound unique.
// Pitfall 10: validateTalabatXlsxHeaders runs before any row parsing.

import * as XLSX from "xlsx";
import { prisma } from "../../../config";
import {
  AdapterSource, DateRange, IngestAdapter, NormalizedRow, NotAvailable, Platform, XlsxIngestResult,
} from "../types";
import { validateTalabatXlsxHeaders, parseTalabatRow, REQUIRED_TALABAT_COLUMNS } from "./xlsxSchema";

export class TalabatXlsxAdapter implements IngestAdapter {
  readonly platform: Platform = "TALABAT";
  readonly source: AdapterSource = "XLSX_IMPORT";

  async isAvailable(_tenantId: string): Promise<boolean> { return true; }

  async fetchOrders(_tenantId: string, _range: DateRange): Promise<NormalizedRow<unknown>[]> {
    throw new NotAvailable("XLSX adapter is upload-driven; use ingestXlsx");
  }

  async ingestXlsx(tenantId: string, buffer: Buffer): Promise<XlsxIngestResult> {
    const wb = XLSX.read(buffer, { type: "buffer" });
    const sheet = wb.Sheets[wb.SheetNames[0]];
    const aoa: any[][] = XLSX.utils.sheet_to_json(sheet, { header: 1 });
    if (aoa.length < 2) {
      throw new Error("Talabat XLSX has no data rows");
    }
    const headers = aoa[0].map((h) => String(h ?? "").trim());
    validateTalabatXlsxHeaders(headers);

    const rowsIn = aoa.length - 1;
    let rowsOk = 0;
    const errors: string[] = [];

    for (let i = 1; i < aoa.length; i++) {
      const arr = aoa[i];
      if (!arr || arr.every((c) => c === null || c === undefined || c === "")) continue;
      try {
        const row = parseTalabatRow(headers, arr);
        const driver = await prisma.driver.findFirst({
          where: { tenantId, platformDriverId: row.driverId, platform: "TALABAT" },
        });
        if (!driver) {
          errors.push(`Driver not found: ${row.driverId} (row ${i + 1})`);
          continue;
        }
        // TalabatDailyMetrics @@unique constraint — verify exact key in schema.prisma during execution
        // and adjust upsert.where shape; the planner expects something like tenantId_driverId_shiftDate
        // or tenantId_driverId_date — the executor reads the schema and matches.
        await prisma.talabatDailyMetrics.upsert({
          where: {
            tenantId_driverId_shiftDate: {
              tenantId,
              driverId: driver.id,
              shiftDate: row.date,
            },
          },
          create: {
            tenantId, driverId: driver.id, shiftDate: row.date,
            ordersDelivered: row.ordersCount,
            onlineMinutes: row.onlineMinutes,
            // Map attendance_status into existing enum/string column on the model — read schema first
          },
          update: {
            ordersDelivered: row.ordersCount,
            onlineMinutes: row.onlineMinutes,
          },
        });
        rowsOk++;
      } catch (err: any) {
        errors.push(`Row ${i + 1}: ${err.message}`);
      }
    }
    return { rowsIn, rowsOk, errors };
  }
}
```

```typescript
// Wraps existing parseAmericanaDailyXlsx + delegates to existing

import { parseAmericanaDailyXlsx } from "../../americanaDailyParser";
import { processIngestionRows } from "../../../queues/americanaIngestWorker";
import {
  AdapterSource, DateRange, IngestAdapter, NormalizedRow, NotAvailable, Platform, XlsxIngestResult,
} from "../types";

export class AmericanaXlsxAdapter implements IngestAdapter {
  readonly platform: Platform = "AMERICANA";
  readonly source: AdapterSource = "XLSX_IMPORT";

  async isAvailable(_tenantId: string): Promise<boolean> { return true; }

  async fetchOrders(_tenantId: string, _range: DateRange): Promise<NormalizedRow<unknown>[]> {
    throw new NotAvailable("XLSX adapter is upload-driven; use ingestXlsx");
  }

  async ingestXlsx(tenantId: string, buffer: Buffer): Promise<XlsxIngestResult> {
    const rows = parseAmericanaDailyXlsx(buffer);
    if (rows.length === 0) {
      return { rowsIn: 0, rowsOk: 0, errors: ["XLSX produced 0 rows"] };
    }
    // Stage AmericanaDailyIngestion row matching existing route pattern.
    const ingestion = await prisma.americanaDailyIngestion.create({
      data: {
        tenantId,
        sourceFile: `xlsx-import-${Date.now()}.xlsx`,
        status: "PENDING",
        rowCount: rows.length,
      },
      select: { id: true },
    });
    // Delegate to existing pipeline so violation + attendance side-effects fire.
    const result = await processIngestionRows({
      tenantId,
      ingestionId: ingestion.id,
      rows,
    });
    return {
      rowsIn: rows.length,
      rowsOk: result.rowsOk,
      errors: result.errors,
    };
  }
}
```

```typescript
// Thin re-export shim per orchestrator resolution #5. Existing IMAP watcher
// at services/americanaInboxWatcher.ts stays where it is; this adapter just
// exposes it through the IngestAdapter interface so the registry composes

import { prisma } from "../../../config";
import { pollTenantInbox } from "../../americanaInboxWatcher";
import {
  AdapterSource, DateRange, IngestAdapter, NormalizedRow, NotAvailable, Platform,
} from "../types";

interface InboxConfig {
  host: string; port?: number; secure?: boolean; user: string; password: string; mailbox?: string;
}

function readInboxConfig(settings: any): InboxConfig | null {
  const cfg = settings?.americana?.ingest;
  if (!cfg?.host || !cfg?.user || !cfg?.password) return null;
  return {
    host: cfg.host, port: cfg.port ?? 993, secure: cfg.secure !== false,
    user: cfg.user, password: cfg.password, mailbox: cfg.mailbox ?? "INBOX",
  };
}

export class AmericanaEmailAdapter implements IngestAdapter {
  readonly platform: Platform = "AMERICANA";
  readonly source: AdapterSource = "EMAIL_INBOX";

  async isAvailable(tenantId: string): Promise<boolean> {
    const tenant = await prisma.tenant.findUnique({
      where: { id: tenantId }, select: { settings: true },
    });
    return readInboxConfig(tenant?.settings) !== null;
  }

  /** Manual trigger — returns count of newly ingested messages. Wave 4's
   *  backwash pullChunkPhase6 may invoke this; production schedulers continue
   *  to use the existing americanaInboxWatcher polling loop. */
  async run(tenantId: string): Promise<number> {
    const tenant = await prisma.tenant.findUnique({
      where: { id: tenantId }, select: { settings: true },
    });
    const cfg = readInboxConfig(tenant?.settings);
    if (!cfg) throw new NotAvailable("No Americana inbox config for tenant");
    return pollTenantInbox(tenantId, cfg);
  }

  async fetchOrders(_tenantId: string, _range: DateRange): Promise<NormalizedRow<unknown>[]> {
    throw new NotAvailable("Email is push-driven; use run() to poll inbox");
  }
}
```
