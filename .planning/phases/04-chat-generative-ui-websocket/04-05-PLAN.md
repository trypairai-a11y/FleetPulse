**Step 1 — Build `backend/src/queues/scheduledBriefingsWorker.ts`:**

```typescript
import { Queue, Worker, JobScheduler } from "bullmq";
import { redisConnection } from "../services/eventBus";
import { runAgent } from "../agent";
import { recordMetricEvent } from "../agent/metricEvent";
import * as chatHistoryService from "../services/chatHistoryService";
import { prisma } from "../config";
import { logger } from "../config/logger";

const QUEUE = "scheduled-briefings";

export const scheduledBriefingsQueue = new Queue(QUEUE, { connection: redisConnection });
const scheduledBriefingsScheduler = new JobScheduler(QUEUE, { connection: redisConnection });

export async function bindBriefing(briefing: {
  id: string; tenantId: string; userId: string;
  cron: string; prompt: string; type: string;
}) {
  await scheduledBriefingsScheduler.upsertJobScheduler(
    `sb:${briefing.id}`,
    { pattern: briefing.cron, tz: "Asia/Kuwait" },   // RESEARCH Pitfall 7 — Kuwait no DST
    {
      name: "scheduled-briefing-tick",
      data: {
        briefingId: briefing.id, tenantId: briefing.tenantId, userId: briefing.userId,
        prompt: briefing.prompt, type: briefing.type,
      },
    },
  );
}

export async function unbindBriefing(briefingId: string) {
  await scheduledBriefingsScheduler.removeJobScheduler(`sb:${briefingId}`);
}

export const scheduledBriefingsWorker = new Worker(
  QUEUE,
  async (job) => {
    const { briefingId, tenantId, userId, prompt, type } = job.data as any;

    if (type === "standing_rule_v3") {
      logger.info({ briefingId }, "scheduledBriefingsWorker: standing_rule_v3 no-op (Phase 12 wires firing)");
      await recordMetricEvent({
        tenantId, userId, eventType: "standing_rule_skipped",
        properties: { briefingId, reason: "phase_4_no_op" },
      });
      return;
    }

    if (type !== "briefing") {
      logger.warn({ briefingId, type }, "scheduledBriefingsWorker: unknown type — skipping");
      return;
    }

    const result = await runAgent("chat", {
      tenantId,
      triggerEvent: `cron:scheduled-briefing:${briefingId}`,
      userMessage: prompt,
    });

    if (result.status !== "completed") {
      logger.warn({ briefingId, status: result.status, error: result.error }, "scheduledBriefingsWorker: agent run failed");
      await recordMetricEvent({
        tenantId, userId, eventType: "scheduled_briefing_failed",
        properties: { briefingId, runId: result.runId, status: result.status, error: result.error },
      });
      return;
    }

    // Persist briefing as a new chat thread + 2 messages.
    const thread = await chatHistoryService.upsertThread({
      tenantId, userId, source: "briefing", briefingId, firstMessage: prompt,
    });
    await chatHistoryService.appendMessage({
      threadId: thread.id, role: "user", content: prompt, state: "complete",
    });
    await chatHistoryService.appendMessage({
      threadId: thread.id, role: "assistant",
      content: result.text ?? "",
      views: (result.views ?? []) as any[],
      state: "complete",
    });

    // Update lastFireAt; nextFireAt is computed by BullMQ.
    await prisma.scheduledBriefing.updateMany({
      where: { id: briefingId, tenantId, userId },
      data: { lastFireAt: new Date() },
    });

    await recordMetricEvent({
      tenantId, userId, eventType: "scheduled_briefing_fired",
      properties: { briefingId, runId: result.runId, threadId: thread.id, viewCount: (result.views ?? []).length },
    });
  },
  { connection: redisConnection },
);
```

```typescript
import { prisma } from "../config";
import { bindBriefing, unbindBriefing } from "../queues/scheduledBriefingsWorker";

const ALLOWED_CRON_DEFAULT = new Set(["0 6 * * *", "0 7 * * *"]);

export function validateCron(cron: string, userRole: string): { ok: boolean; error?: string } {
  if (ALLOWED_CRON_DEFAULT.has(cron)) return { ok: true };
  if (userRole === "ADMIN") {
    // Best-effort regex sanity for admin custom: 5 fields. Reject `* * * * *` and other sub-5-min schedules.
    const fields = cron.trim().split(/\s+/);
    if (fields.length !== 5) return { ok: false, error: "Cron must have 5 fields" };
    if (fields[0] === "*") return { ok: false, error: "Sub-5-min schedules are not allowed" };
    return { ok: true };
  }
  return { ok: false, error: "Custom cron requires ADMIN role" };
}

export async function createBriefing(opts: { tenantId: string; userId: string; userRole: string; name: string; cron: string; prompt: string; recipients?: string[]; channels?: string[]; type?: "briefing"|"standing_rule_v3"; }) {
  const valid = validateCron(opts.cron, opts.userRole);
  if (!valid.ok) throw new Error(valid.error);
  const created = await prisma.scheduledBriefing.create({
    data: {
      tenantId: opts.tenantId, userId: opts.userId,
      name: opts.name, cron: opts.cron, prompt: opts.prompt,
      recipients: (opts.recipients ?? []) as any,
      channels: (opts.channels ?? ["in_chat"]) as any,
      type: opts.type ?? "briefing",
      active: true,
    },
  });
  if (created.active) await bindBriefing(created);
  return created;
}

export async function listBriefings(tenantId: string, userId: string) {
  return prisma.scheduledBriefing.findMany({
    where: { tenantId, userId },
    orderBy: { createdAt: "desc" },
  });
}

export async function patchBriefing(id: string, tenantId: string, userId: string, userRole: string, patch: Partial<{ active: boolean; name: string; cron: string; prompt: string }>) {
  if (patch.cron) {
    const valid = validateCron(patch.cron, userRole);
    if (!valid.ok) throw new Error(valid.error);
  }
  const updated = await prisma.scheduledBriefing.updateMany({
    where: { id, tenantId, userId }, data: patch,
  });
  if (updated.count === 0) throw new Error("Briefing not found");
  const fresh = await prisma.scheduledBriefing.findFirst({ where: { id, tenantId, userId } });
  if (!fresh) throw new Error("Briefing not found");
  if (fresh.active) await bindBriefing(fresh);
  else await unbindBriefing(fresh.id);
  return fresh;
}

export async function deleteBriefing(id: string, tenantId: string, userId: string) {
  await unbindBriefing(id);
  await prisma.scheduledBriefing.deleteMany({ where: { id, tenantId, userId } });
}
```
