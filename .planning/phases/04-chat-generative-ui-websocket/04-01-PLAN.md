```prisma
// ─── Phase 4 Wave 1 — Chat surface + scheduled briefings ───
// REQ-chat-global-access + REQ-chat-generated-dashboards + REQ-chat-scheduled-jobs.
// Tenant-scoped + user-scoped (defense in depth — see RESEARCH §"Per-User
// Searchable Chat History"). 90-day retention via archivedAt; chatHistoryService
// runs the cleanup. Standing-rule type kept on ScheduledBriefing for forward
// compatibility — Phase 12 (REQ-data-agent-rule) wires the firing engine.

model ChatThread {
  id            String    @id @default(cuid())
  tenantId      String
  userId        String
  title         String    @default("New chat")
  pinned        Boolean   @default(false)
  source        String    @default("user")        // "user" | "briefing"
  briefingId    String?                            // FK to ScheduledBriefing if source="briefing"
  lastMessageAt DateTime  @default(now())
  archivedAt    DateTime?                          // 90-day retention; cleanup job sets this
  createdAt     DateTime  @default(now())
  updatedAt     DateTime  @updatedAt

  tenant        Tenant              @relation(fields: [tenantId], references: [id])
  user          User                @relation("ChatThreadOwner", fields: [userId], references: [id])
  briefing      ScheduledBriefing?  @relation(fields: [briefingId], references: [id], onDelete: SetNull)
  messages      ChatMessage[]

  @@index([tenantId, userId, lastMessageAt(sort: Desc)])
  @@index([userId, pinned, lastMessageAt(sort: Desc)])
  @@index([tenantId, archivedAt])
}

model ChatMessage {
  id               String    @id @default(cuid())
  tenantId         String
  threadId         String
  role             String                            // "user" | "assistant" | "system"
  content          String    @db.Text
  views            Json      @default("[]")          // GeneratedView[]
  toolCalls        Json      @default("[]")          // ToolCallRecord[]
  proposalId       String?                           // FK to AgentAction; assistant turns may carry one
  state            String    @default("complete")    // "queued"|"tool_running"|"streaming"|"complete"|"error"|"cancelled"
  promptTokens     Int       @default(0)
  completionTokens Int       @default(0)
  latencyMs        Int       @default(0)
  modelName        String?
  errorMessage     String?
  createdAt        DateTime  @default(now())

  tenant   Tenant       @relation(fields: [tenantId], references: [id])
  thread   ChatThread   @relation(fields: [threadId], references: [id], onDelete: Cascade)
  proposal AgentAction? @relation("ChatMessageProposal", fields: [proposalId], references: [id], onDelete: SetNull)

  @@index([threadId, createdAt])
  @@index([tenantId, createdAt])
}

model ScheduledBriefing {
  id          String     @id @default(cuid())
  tenantId    String
  userId      String                                  // creator
  name        String
  cron        String                                  // validated whitelist; "0 6 * * *" / "0 7 * * *" / admin custom
  prompt      String     @db.Text
  recipients  Json       @default("[]")               // string[] of userIds
  channels    Json       @default("[\"in_chat\"]")    // ("in_chat"|"email")[]
  type        String     @default("briefing")         // "briefing" | "standing_rule_v3" — standing rule UI-only in Phase 4
  active      Boolean    @default(true)
  nextFireAt  DateTime?
  lastFireAt  DateTime?
  createdAt   DateTime   @default(now())
  updatedAt   DateTime   @updatedAt

  tenant   Tenant       @relation(fields: [tenantId], references: [id])
  creator  User         @relation("ScheduledBriefingCreator", fields: [userId], references: [id])
  threads  ChatThread[]

  @@index([tenantId, active, nextFireAt])
  @@index([userId, active])
}
```
