```typescript
import { api } from "./index";   // existing axios instance with cookie auth
import type { ChatThread, ChatMessage } from "@/types/chat";

export const chatApi = {
  async createThread(initialMessage?: string): Promise<{ thread: ChatThread }> {
    return (await api.post("/api/chat/threads", { initialMessage })).data;
  },
  async listThreads(opts?: { search?: string; page?: number; limit?: number }) {
    return (await api.get("/api/chat/threads", { params: opts })).data as {
      threads: ChatThread[];
      pagination: { page: number; limit: number; total: number; totalPages: number };
      grouped?: { today: ChatThread[]; yesterday: ChatThread[]; thisWeek: ChatThread[]; older: ChatThread[] };
      search?: { matches: any[] };
    };
  },
  async getThread(id: string) {
    return (await api.get(`/api/chat/threads/${id}`)).data as { thread: ChatThread; messages: ChatMessage[]; hasMoreMessages: boolean };
  },
  async patchThread(id: string, patch: { title?: string; pinned?: boolean }) {
    return (await api.patch(`/api/chat/threads/${id}`, patch)).data as { thread: ChatThread };
  },
  async deleteThread(id: string) {
    return (await api.delete(`/api/chat/threads/${id}`)).data;
  },
  async sendMessage(threadId: string, content: string) {
    return (await api.post(`/api/chat/threads/${threadId}/messages`, { content })).data as {
      userMessage: ChatMessage; assistantMessageId: string;
    };
  },
  async cancelMessage(messageId: string) {
    return (await api.post(`/api/chat/messages/${messageId}/cancel`)).data;
  },
};
```

**Step 2 — Build `frontend/src/hooks/useStreamingChat.ts`:**

```typescript
import { useEffect, useRef, useState } from "react";
import type { GeneratedView } from "@/types/chat";

interface UseStreamingChatOpts {
  threadId?: string;
  userMessage?: string;
  onTextDelta?(delta: string): void;
  onViewBlock?(view: GeneratedView): void;
  onProposal?(pendingActionId: string): void;
  onComplete?(meta: { runId: string; promptTokens: number; completionTokens: number; latencyMs: number }): void;
  onError?(error: string): void;
  onCancelled?(): void;
  autoStart?: boolean;     // default true; set false for permalink view
}

export function useStreamingChat(opts: UseStreamingChatOpts) {
  const [isConnected, setConnected] = useState(false);
  const [isStreaming, setStreaming] = useState(false);
  const sourceRef = useRef<EventSource | null>(null);
  const retryCountRef = useRef(0);
  const retryTimeoutRef = useRef<NodeJS.Timeout | null>(null);

  const connect = () => {
    if (!opts.userMessage) return;   // nothing to send
    const params = new URLSearchParams();
    if (opts.threadId) params.set("threadId", opts.threadId);
    params.set("q", opts.userMessage);
    const url = `/api/ai/chat/stream?${params.toString()}`;

    const src = new EventSource(url, { withCredentials: true });
    sourceRef.current = src;
    setStreaming(true);

    src.addEventListener("open", () => {
      setConnected(true);
      retryCountRef.current = 0;
    });

    src.addEventListener("thread", (e: MessageEvent) => {
      // server's first event after auth — confirms threadId
      const data = JSON.parse(e.data);
      // (caller may track via onComplete or expose as state)
    });
    src.addEventListener("queued", () => { /* set initial state if needed */ });
    src.addEventListener("text_delta", (e: MessageEvent) => {
      const { delta } = JSON.parse(e.data);
      opts.onTextDelta?.(delta);
    });
    src.addEventListener("view_block", (e: MessageEvent) => {
      const { view } = JSON.parse(e.data);
      opts.onViewBlock?.(view);
    });
    src.addEventListener("proposal", (e: MessageEvent) => {
      const { pendingActionId } = JSON.parse(e.data);
      opts.onProposal?.(pendingActionId);
    });
    src.addEventListener("complete", (e: MessageEvent) => {
      const data = JSON.parse(e.data);
      opts.onComplete?.(data.meta ?? data);
      setStreaming(false);
      src.close();
    });
    src.addEventListener("cancelled", () => {
      opts.onCancelled?.();
      setStreaming(false);
      src.close();
    });
    src.addEventListener("error", (e: MessageEvent) => {
      // Browser EventSource error fires on disconnect AND on server-sent error events.
      // Differentiate by readyState:
      if (src.readyState === EventSource.CLOSED) {
        // server-side error or stream end
        opts.onError?.(typeof e.data === "string" ? e.data : "stream closed");
        setStreaming(false);
      } else if (src.readyState === EventSource.CONNECTING && retryCountRef.current < 3) {
        // browser auto-reconnects, but cap retries
        const delay = Math.min(30_000, 1_000 * Math.pow(2, retryCountRef.current));
        retryCountRef.current += 1;
        retryTimeoutRef.current = setTimeout(() => connect(), delay);
        src.close();
      } else {
        opts.onError?.("connection lost");
        setStreaming(false);
        src.close();
      }
    });
  };

  const abort = () => {
    sourceRef.current?.close();
    sourceRef.current = null;
    setStreaming(false);
  };

  const retry = () => {
    abort();
    retryCountRef.current = 0;
    connect();
  };

  useEffect(() => {
    if (opts.autoStart !== false && opts.userMessage) connect();
    return () => {
      abort();
      if (retryTimeoutRef.current) clearTimeout(retryTimeoutRef.current);
    };
    // eslint-disable-next-line react-hooks/exhaustive-deps
  }, []);

  return { isConnected, isStreaming, abort, retry };
}
```

```typescript
export function useChatThread(id: string | null) {

    queryKey: ["chat-thread", id],
```

```tsx
"use client";
import { useState } from "react";
import { Pin } from "lucide-react";
import { api } from "@/lib/api";
import type { GeneratedView } from "@/types/chat";

export function PinButton({ view, threadId, messageId }: { view: GeneratedView; threadId?: string; messageId?: string }) {
  const [pinned, setPinned] = useState(!!view.pinnedViewId);
  const [busy, setBusy] = useState(false);
  const onPin = async () => {
    setBusy(true);
    try {
      await api.post("/api/pinned-views", {
        viewType: view.viewType,
        spec: view.spec,
        title: view.title ?? `${view.viewType} pinned`,
        sortOrder: 9999,
        refreshFrequency: "on_open",
        sourceThreadId: threadId,
        sourceMessageId: messageId,
      });
      setPinned(true);
    } catch (err) { /* toast */ }
    finally { setBusy(false); }
  };
  return (
    <button onClick={onPin} disabled={pinned || busy}
      className="absolute right-3 top-3 inline-flex items-center gap-1 text-[11px] text-secondary hover:text-primary disabled:opacity-50">
      <Pin className="h-3 w-3" /> {pinned ? "Pinned" : "Pin to Home"}
    </button>
  );
}
```

**Step 4 — Build ChatActionCard:**

```tsx
"use client";
import { useState, useEffect } from "react";
import { useQuery } from "@tanstack/react-query";
import { api } from "@/lib/api";
import { TagPill } from "@/components/decisions/TagPill";
import { EvidenceList } from "@/components/decisions/EvidenceList";

export function ChatActionCard({ proposalId, threadId, messageId }: {
  proposalId: string; threadId: string; messageId: string;
}) {
  const [state, setState] = useState<"pending"|"approving"|"approved"|"dismissed"|"error">("pending");
  const [error, setError] = useState<string | null>(null);
  const { data: action } = useQuery({
    queryKey: ["pending-action", proposalId],
    queryFn: () => api.get(`/api/decisions/${proposalId}`).then(r => r.data),
  });
  if (!action) return null;

  const onApprove = async () => {
    setState("approving");
    try {
      await api.post(`/api/decisions/${proposalId}/approve`, { source: "chat", threadId, msgId: messageId });
      setState("approved");
      // 5s undo window — set timeout; clearable via Undo click
      setTimeout(() => { /* finalize: card collapses */ }, 5000);
    } catch (err: any) {
      if (err.response?.status === 409) {
        setError("Already resolved in /decisions.");
        setState("error");
      } else {
        setError(err.message);
        setState("pending");
      }
    }
  };
  const onDismiss = async () => {
    try { await api.post(`/api/decisions/${proposalId}/dismiss`, { reason: "via chat" }); setState("dismissed"); }
    catch (err: any) { setError(err.message); setState("pending"); }
  };
  const onUndo = async () => {
    try { await api.post(`/api/decisions/${proposalId}/undo`); setState("pending"); }
    catch (err: any) { setError(err.message); }
  };

  if (state === "approved") {
    return (
      <div className="rounded-xl ring-1 ring-primary/30 bg-primary/5 p-4 mb-3">
        <span className="text-sm text-primary">Approved · </span>
        <button onClick={onUndo} className="text-sm text-primary underline">Undo</button>
      </div>
    );
  }

  return (
    <div className="rounded-xl ring-1 ring-sand-200 bg-card p-4 mb-3">
      <div className="flex items-center gap-2 mb-2">
        <TagPill tag={action.tag} />
        <span className="text-sm font-medium">{action.headline}</span>
      </div>
      <p className="text-xs text-secondary line-clamp-1 mb-3">{action.reasoning}</p>
      {/* Optional disclosures from EvidenceList */}
      <div className="flex gap-2 mt-3">
        <button onClick={onApprove} disabled={state === "approving"}
          className="rounded-lg bg-foreground text-white px-3 py-1.5 text-xs font-medium disabled:opacity-50">
          Approve
        </button>
        <button onClick={() => { /* open EditDrawer modal — pass proposal */ }}
          className="rounded-lg ring-1 ring-sand-200 px-3 py-1.5 text-xs font-medium">
          Edit
        </button>
        <button onClick={onDismiss}
          className="rounded-lg ring-1 ring-sand-200 px-3 py-1.5 text-xs font-medium text-secondary">
          Dismiss
        </button>
      </div>
      {error && <p className="text-xs text-red-600 mt-2">{error}</p>}
    </div>
  );
}
```
