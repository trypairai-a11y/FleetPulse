**Step 1 — Build `backend/src/routes/pinnedViews.ts`:**

```typescript
import { Router, Request, Response } from "express";
import { authMiddleware } from "../middleware/auth";
import { tenantScope } from "../middleware/tenantScope";
import { prisma } from "../config";
import { createPinnedView, listPinsForUser, removePinnedView } from "../agent/pinnedView";
import { runAgent } from "../agent";
import { logger } from "../config/logger";

const router = Router();
router.use(authMiddleware, tenantScope);

const SOFT_CAP = 24;

router.post("/", async (req: Request, res: Response) => {
  const userId = (req.user as any).id as string;
  const tenantId = (req.user as any).tenantId as string;
  const { viewType, spec, title, description, sortOrder = 0,
          refreshFrequency = "on_open", sourceThreadId, sourceMessageId } = req.body ?? {};
  if (!viewType || !spec || !title) {
    res.status(400).json({ error: "viewType, spec, title required" }); return;
  }
  // Soft cap warning (frontend can override; backend warns):
  const existing = await prisma.pinnedView.count({ where: { tenantId, userId } });
  const warnSoftCap = existing >= SOFT_CAP;

  // Dedupe per RESEARCH "Anti-Patterns": same (userId, sourceMessageId, viewType, title) returns existing.
  if (sourceMessageId) {
    const dup = await prisma.pinnedView.findFirst({
      where: { tenantId, userId, sourceMessageId, viewType, title },
    });
    if (dup) {
      res.json({ pinnedView: dup, deduplicated: true, warnSoftCap });
      return;
    }
  }

  const created = await createPinnedView({
    tenantId, userId, viewType, spec, title, description, sortOrder,
    refreshFrequency, sourceThreadId, sourceMessageId,
  });
  const full = await prisma.pinnedView.findFirst({ where: { id: created.id, tenantId, userId } });
  res.json({ pinnedView: full, warnSoftCap });
});

router.get("/", async (req: Request, res: Response) => {
  const userId = (req.user as any).id as string;
  const tenantId = (req.user as any).tenantId as string;
  const pins = await listPinsForUser(tenantId, userId);
  res.json({ pinnedViews: pins });
});

router.patch("/:id", async (req: Request, res: Response) => {
  const userId = (req.user as any).id as string;
  const tenantId = (req.user as any).tenantId as string;
  const { title, description, sortOrder, refreshFrequency } = req.body ?? {};
  const allowed: Record<string, unknown> = {};
  if (typeof title === "string") allowed.title = title.slice(0, 200);
  if (typeof description === "string") allowed.description = description;
  if (typeof sortOrder === "number") allowed.sortOrder = sortOrder;
  if (["on_open","live","static"].includes(refreshFrequency)) allowed.refreshFrequency = refreshFrequency;
  const updated = await prisma.pinnedView.updateMany({
    where: { id: req.params.id, tenantId, userId }, data: allowed,
  });
  if (updated.count === 0) { res.status(404).json({ error: "PinnedView not found" }); return; }
  const full = await prisma.pinnedView.findFirst({ where: { id: req.params.id, tenantId, userId } });
  res.json({ pinnedView: full });
});

router.delete("/:id", async (req: Request, res: Response) => {
  const userId = (req.user as any).id as string;
  const tenantId = (req.user as any).tenantId as string;
  const { removed } = await removePinnedView(req.params.id, tenantId, userId);
  if (!removed) { res.status(404).json({ error: "PinnedView not found" }); return; }
  res.json({ ok: true });
});

router.post("/:id/refresh", async (req: Request, res: Response) => {
  const userId = (req.user as any).id as string;
  const tenantId = (req.user as any).tenantId as string;
  const pin = await prisma.pinnedView.findFirst({
    where: { id: req.params.id, tenantId, userId },
  });
  if (!pin) { res.status(404).json({ error: "PinnedView not found" }); return; }
  // Refresh requires a sourceMessageId or sourcePrompt to re-run.
  // Phase 4: read the original prompt from chat history (via sourceMessageId → ChatMessage.content of the
  // matching user message in the thread). If we can't find it, return the existing spec unchanged.
  const sourceMsg = pin.sourceMessageId
    ? await prisma.chatMessage.findFirst({ where: { id: pin.sourceMessageId, tenantId } })
    : null;
  if (!sourceMsg) { res.json({ pinnedView: pin, refreshedSpec: pin.spec }); return; }
  // The user message that originated this pin lives BEFORE the assistant message in the thread.
  // Find the closest user message:
  const userMsg = await prisma.chatMessage.findFirst({
    where: {
      threadId: sourceMsg.threadId, tenantId, role: "user",
      createdAt: { lt: sourceMsg.createdAt },
    },
    orderBy: { createdAt: "desc" },
  });
  if (!userMsg) { res.json({ pinnedView: pin, refreshedSpec: pin.spec }); return; }

  const result = await runAgent("chat", {
    tenantId, triggerEvent: "route:pinned-views:refresh",
    userMessage: userMsg.content,
  });

  // Pick the first view of the same viewType as the pin (best-effort match).
  const fresh = (result.views ?? []).find((v: any) => v?.type === pin.viewType || v?.viewType === pin.viewType);
  if (!fresh) { res.json({ pinnedView: pin, refreshedSpec: pin.spec, note: "no matching view in re-run" }); return; }

  await prisma.pinnedView.update({
    where: { id: pin.id },
    data: {
      spec: fresh as any,
      lastViewedAt: new Date(),
    },
  });
  const updated = await prisma.pinnedView.findFirst({ where: { id: pin.id, tenantId, userId } });
  res.json({ pinnedView: updated, refreshedSpec: fresh });
});

export default router;
```

```typescript
import { api } from "./index";
import type { PinnedView } from "@/types/chat";   // Add to chat.ts: PinnedView interface mirroring backend

export const pinnedViewsApi = {
  async list() {
    return (await api.get("/api/pinned-views")).data as { pinnedViews: PinnedView[] };
  },
  async create(body: {
    viewType: string; spec: object; title: string;
    description?: string; sortOrder?: number;
    refreshFrequency?: "on_open"|"live"|"static";
    sourceThreadId?: string; sourceMessageId?: string;
  }) {
    return (await api.post("/api/pinned-views", body)).data as {
      pinnedView: PinnedView; warnSoftCap?: boolean; deduplicated?: boolean;
    };
  },
  async patch(id: string, body: Partial<{ title: string; description: string; sortOrder: number; refreshFrequency: "on_open"|"live"|"static" }>) {
    return (await api.patch(`/api/pinned-views/${id}`, body)).data;
  },
  async remove(id: string) {
    return (await api.delete(`/api/pinned-views/${id}`)).data;
  },
  async refresh(id: string) {
    return (await api.post(`/api/pinned-views/${id}/refresh`)).data;
  },
};
```

```typescript
import { useQuery, useMutation, useQueryClient } from "@tanstack/react-query";
import { pinnedViewsApi } from "@/lib/api/pinnedViews";

const KEY = ["pinned-views"];

export function usePinnedViews() {
  return useQuery({
    queryKey: KEY, queryFn: pinnedViewsApi.list,
    refetchInterval: 30_000,        // every 30s — live tiles see updates
    staleTime: 25_000,
  });
}

export function useUnpinView() {
  const qc = useQueryClient();
  return useMutation({
    mutationFn: (id: string) => pinnedViewsApi.remove(id),
    onSuccess: () => qc.invalidateQueries({ queryKey: KEY }),
  });
}

export function useRefreshPin() {
  const qc = useQueryClient();
  return useMutation({
    mutationFn: (id: string) => pinnedViewsApi.refresh(id),
    onSuccess: () => qc.invalidateQueries({ queryKey: KEY }),
  });
}
```

**Step 4 — Build `frontend/src/components/decisions/PinnedViewsRail.tsx`:**

```tsx
"use client";
import { useEffect, useState } from "react";
import { ChevronDown, ChevronRight } from "lucide-react";
import { usePinnedViews } from "@/hooks/usePinnedViews";
import { PinnedViewTile } from "./PinnedViewTile";

const STORAGE_KEY = "darb.decisions.pinnedRailCollapsed";

export function PinnedViewsRail() {
  const { data, isLoading } = usePinnedViews();
  const [collapsed, setCollapsed] = useState(true);
  useEffect(() => {
    const v = localStorage.getItem(STORAGE_KEY);
    if (v !== null) setCollapsed(v === "true");
  }, []);
  const toggle = () => {
    const next = !collapsed;
    setCollapsed(next);
    localStorage.setItem(STORAGE_KEY, String(next));
  };
  if (isLoading || !data) return null;
  const pins = data.pinnedViews;
  if (pins.length === 0) return null;   // Empty state hides rail entirely (UI-SPEC §3.3.4)

  return (
    <section className="mb-6 rounded-2xl ring-1 ring-sand-200 bg-card p-4">
      <header className="flex items-center justify-between mb-3">
        <div className="flex items-center gap-2">
          <button onClick={toggle} className="inline-flex items-center gap-1 text-sm font-medium">
            {collapsed ? <ChevronRight className="h-4 w-4" /> : <ChevronDown className="h-4 w-4" />}
            Pinned views ({pins.length})
          </button>
        </div>
      </header>
      {!collapsed && (
        <div className="flex gap-3 overflow-x-auto pb-2">
          {pins.slice(0, 6).map(p => <PinnedViewTile key={p.id} pin={p} />)}
          {pins.length > 6 && (
            <button className="text-sm text-primary self-center">View all ({pins.length}) →</button>
          )}
        </div>
      )}
    </section>
  );
}
```
