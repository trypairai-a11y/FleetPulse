**Step 1 — Build `backend/src/services/chatStreamService.ts`:**

The service owns the SSE write protocol, heartbeat scheduling, Redis lock acquisition, and the runAgent invocation. The route handler stays thin.

```typescript
import type { Request, Response } from "express";
import { runAgent } from "../agent";
import * as chatHistoryService from "./chatHistoryService";
import { redisConnection } from "./eventBus";
import { logger } from "../config/logger";

const HEARTBEAT_MS = 15_000;
const LOCK_TTL_MS = 60_000;
const LOCK_PREFIX = "chat:lock:";
const CANCEL_PREFIX = "chat:cancel:";

export async function acquireThreadLock(threadId: string, ttlMs = LOCK_TTL_MS) {
  const key = LOCK_PREFIX + threadId;
  const ok = await redisConnection.set(key, "1", "PX", ttlMs, "NX");
  if (!ok) return null;
  return {
    async released() { await redisConnection.del(key); },
  };
}

async function setCancelFlag(messageId: string) {
  await redisConnection.set(CANCEL_PREFIX + messageId, "1", "PX", LOCK_TTL_MS);
}

async function isCancelled(messageId: string): Promise<boolean> {
  return (await redisConnection.exists(CANCEL_PREFIX + messageId)) === 1;
}

export async function cancelStream(messageId: string) {
  await setCancelFlag(messageId);
}

export async function streamChatToClient(opts: {
  tenantId: string; userId: string;
  threadId?: string; userMessage: string;
  res: Response;
}) {
  const { tenantId, userId, userMessage, res } = opts;

  // SSE setup. X-Accel-Buffering disables Vercel/nginx buffering on the response.
  res.setHeader("Content-Type", "text/event-stream");
  res.setHeader("Cache-Control", "no-cache, no-transform");
  res.setHeader("Connection", "keep-alive");
  res.setHeader("X-Accel-Buffering", "no");
  res.flushHeaders();

  const send = (eventType: string, data: unknown) => {
    res.write(`event: ${eventType}\ndata: ${JSON.stringify(data)}\n\n`);
  };
  const heartbeat = () => res.write(`:heartbeat\n\n`);
  const heartbeatTimer = setInterval(heartbeat, HEARTBEAT_MS);

  // 1. Persist user message; ensure thread exists.
  const thread = await chatHistoryService.upsertThread({
    tenantId, userId, threadId: opts.threadId, firstMessage: userMessage,
  });
  await chatHistoryService.appendMessage({
    threadId: thread.id, role: "user", content: userMessage, state: "complete",
  });
  send("thread", { threadId: thread.id });

  // 2. Create assistant placeholder message
  const placeholder = await chatHistoryService.appendMessage({
    threadId: thread.id, role: "assistant", content: "", state: "queued",
  });
  send("queued", { msgId: placeholder.id });

  // 3. Acquire per-thread lock to prevent concurrent streams from interleaving.
  const lock = await acquireThreadLock(thread.id);
  if (!lock) {
    send("error", { msgId: placeholder.id, error: "Another query is in flight for this thread." });
    res.end();
    clearInterval(heartbeatTimer);
    return { threadId: thread.id, assistantMessageId: placeholder.id, status: "failed" as const };
  }

  // 4. Listen for client disconnect — cancels mid-stream.
  let clientGone = false;
  const onClose = () => { clientGone = true; };
  res.req.on("close", onClose);

  try {
    const history = await chatHistoryService.recentTurns(thread.id, 20);
    const collectedViews: unknown[] = [];
    let textBuffer = "";
    let firstPendingActionId: string | undefined;

    const result = await runAgent("chat", {
      tenantId,
      triggerEvent: "route:chat:stream",
      userMessage,
      history,
      stream: {
        onTextDelta(delta) {
          textBuffer += delta;
          send("text_delta", { msgId: placeholder.id, delta });
        },
        onView(view) {
          collectedViews.push(view);
          send("view_block", { msgId: placeholder.id, view });
        },
        onPendingAction(pendingActionId) {
          firstPendingActionId ??= pendingActionId;
          send("proposal", { msgId: placeholder.id, pendingActionId });
        },
        onCancel() {
          // poll: client disconnected OR cancel flag set
          return clientGone || /* may be checked sync below */ false;
        },
      },
    });

    // 5. Persist final assistant turn — single update of the placeholder row, not a new row.
    // NB: appendMessage creates new rows; for this we go directly through prisma to update.
    // (Add a helper to chatHistoryService.ts: `finalizeAssistantMessage(messageId, patch)` if desired.)
    // For Wave 2, append a SECOND assistant message with the final state (placeholder remains as "queued"
    // history). Mirrors the simpler pattern; refactor in Phase 11 if needed.

    if (result.status === "cancelled") {
      send("cancelled", { msgId: placeholder.id });
      await chatHistoryService.appendMessage({
        threadId: thread.id, role: "assistant",
        content: textBuffer, views: collectedViews as any[], state: "cancelled",
      });
    } else if (result.status === "completed") {
      const final = await chatHistoryService.appendMessage({
        threadId: thread.id, role: "assistant",
        content: result.text ?? textBuffer,
        views: (result.views ?? collectedViews) as any[],
        toolCalls: [],
        proposalId: firstPendingActionId,
        state: "complete",
        promptTokens: 0, completionTokens: 0, latencyMs: 0,
        modelName: "claude-sonnet-4-6",
      });
      send("complete", {
        msgId: placeholder.id,
        finalMessageId: final.id,
        runId: result.runId,
        meta: {
          promptTokens: 0, completionTokens: 0,
          latencyMs: 0, modelName: "claude-sonnet-4-6",
        },
      });
    } else {
      send("error", { msgId: placeholder.id, error: result.error ?? "Agent failed" });
      await chatHistoryService.appendMessage({
        threadId: thread.id, role: "assistant", content: textBuffer,
        state: "error", errorMessage: result.error,
      });
    }

    res.end();
    return { threadId: thread.id, assistantMessageId: placeholder.id, status: result.status };
  } catch (err: any) {
    logger.error({ err, threadId: thread.id }, "chatStreamService: failed");
    send("error", { error: err.message ?? String(err) });
    res.end();
    return { threadId: thread.id, assistantMessageId: placeholder.id, status: "failed" as const };
  } finally {
    clearInterval(heartbeatTimer);
    res.req.removeListener("close", onClose);
    await lock.released();
  }
}
```

**Step 4 — Heartbeat semantics:**

`HEARTBEAT_MS = 15_000` ms (RESEARCH Pitfall 1). The `setInterval` runs even outside `runAgent` (covers the brief windows when DB writes are happening). Cleanup in `finally`.

**Step 5 — Cancel mechanism:**

`POST /api/chat/messages/:id/cancel` sets a Redis flag with TTL=60s. The `onCancel` callback in `streamChatToClient` reads `clientGone` (HTTP close) but does NOT yet poll Redis — we don't have the messageId until after `appendMessage` returns. Either:
- Pass the placeholder.id via closure (it's available after step 2 above) — store in a local `cancelMsgId` variable; polled in `onCancel` via `await isCancelled(cancelMsgId)`. Note: `onCancel` is synchronous in the existing runtime.ts contract — Wave 1 specified it returns `boolean`. Since Redis ops are async, the simplest fix is: run a setInterval (every 1s) that checks `isCancelled(placeholder.id)` and sets a local `cancelled = true` boolean; `onCancel` returns this boolean. Cleanup in `finally`.

Implement this polling pattern; verify `cancelStream(messageId)` aborts the loop within ~1-2s.

**Why these decisions:**
- The Redis lock is at the THREAD level, not the user level — a user can have many threads in flight, but each thread serializes. RESEARCH Pitfall 2 + UI-SPEC §9 "Sending a message when previous assistant is still streaming" → composer disables; backend lock is the redundant guard.
- HEARTBEAT_MS=15s is RESEARCH Pitfall 1's recommended value; Vercel proxy idle timeout is ~30s.
- Cancel via Redis flag (not in-process state) makes the cancel survive a horizontally-scaled backend (if/when we migrate off Vercel).

function groupByRecency(threads: { lastMessageAt: Date }[]) {
  const now = new Date();
  const yesterday = new Date(now); yesterday.setDate(yesterday.getDate() - 1);
  const weekAgo = new Date(now); weekAgo.setDate(weekAgo.getDate() - 7);
  const out: Record<string, typeof threads> = { today: [], yesterday: [], thisWeek: [], older: [] };
  for (const t of threads) {
    const ts = t.lastMessageAt;
    if (sameDay(ts, now)) out.today.push(t);
    else if (sameDay(ts, yesterday)) out.yesterday.push(t);
    else if (ts > weekAgo) out.thisWeek.push(t);
    else out.older.push(t);
  }
  return out;
}
function sameDay(a: Date, b: Date) {
  return a.getUTCFullYear() === b.getUTCFullYear()
      && a.getUTCMonth() === b.getUTCMonth()
      && a.getUTCDate() === b.getUTCDate();
}
