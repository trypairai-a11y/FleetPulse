      import * as ImageManipulator from "expo-image-manipulator";
      import * as SecureStore from "expo-secure-store";
      import { requestUploadUrl, recordDeliveryPhotoMetadata } from "../api/client";

      export interface UploadDeliveryPhotoArgs {
        orderId: string;
        uri: string;       // Local file URI from expo-camera takePictureAsync
        latitude: number;
        longitude: number;
      }

      export async function uploadDeliveryPhoto(args: UploadDeliveryPhotoArgs): Promise<void> {
        const deviceId = await SecureStore.getItemAsync("device_id");
        if (!deviceId) throw new Error("Device not enrolled");

        // 1. Compress + strip EXIF (re-save discards orientation/EXIF metadata).
        const compressed = await ImageManipulator.manipulateAsync(
          args.uri,
          [{ resize: { width: 1280 } }],
          { compress: 0.7, format: ImageManipulator.SaveFormat.JPEG },
        );

        // 2. Request presigned URL.
        const presign = await requestUploadUrl({
          deviceId,
          orderId: args.orderId,
          contentType: "image/jpeg",
        });

        // 3. Upload bytes directly to R2.
        const blob = await (await fetch(compressed.uri)).blob();
        const putResp = await fetch(presign.url, {
          method: "PUT",
          headers: { "Content-Type": "image/jpeg" },
          body: blob as any,
        });
        if (!putResp.ok) {
          throw new Error(`R2 upload failed: ${putResp.status}`);
        }

        // 4. Record metadata.
        await recordDeliveryPhotoMetadata({
          deviceId,
          orderId: args.orderId,
          key: presign.key,
          capturedAt: new Date().toISOString(),
          latitude: args.latitude,
          longitude: args.longitude,
        });
      }

        const [stage, setStage] = useState<Stage>("explain");

        async function askForeground() {
          setStage("fg-asking");
          const fg = await Location.requestForegroundPermissionsAsync();

            return;
          }
          setStage("bg-explain");

          const result = await startBeacon();
          onComplete(result.ok);
        }

        if (!visible) return null;

        return (
          <Modal visible={visible} animationType="slide" transparent={false}>
            <View style={styles.container}>
              {stage === "explain" && (

                  <Text style={styles.title}>Darb tracks your location during shifts</Text>

                  <TouchableOpacity style={styles.primaryBtn} onPress={askForeground}>
                    <Text style={styles.primaryBtnText}>Continue</Text>
                  </TouchableOpacity>
                  <TouchableOpacity onPress={() => onComplete(false)} style={styles.secondaryBtn}>
                    <Text style={styles.secondaryBtnText}>Not now</Text>

      import { View, Text, StyleSheet } from "react-native";

      export type BeaconStatus = "active" | "paused" | "error";

      export function BatteryStatusBadge({
        beaconStatus,
        batteryLevel,
      }: { beaconStatus: BeaconStatus; batteryLevel: number }) {
        const color =
          beaconStatus === "active" ? "#16a34a" :
          beaconStatus === "paused" ? "#F97316" : "#dc2626";
        const label =
          beaconStatus === "active" ? "GPS Active" :
          beaconStatus === "paused" ? "GPS Paused" : "GPS Error";
        const pct = Math.max(0, Math.min(1, batteryLevel)) * 100;
        return (
          <View style={[styles.badge, { borderColor: color }]}>
            <View style={[styles.dot, { backgroundColor: color }]} />
            <Text style={[styles.label, { color }]}>{label}</Text>
            <Text style={styles.battery}>{pct.toFixed(0)}%</Text>
          </View>
        );
      }

      const styles = StyleSheet.create({
        badge: {
          flexDirection: "row", alignItems: "center", gap: 8,
          borderWidth: 1, borderRadius: 999, paddingHorizontal: 12, paddingVertical: 6,
        },
        dot: { width: 8, height: 8, borderRadius: 4 },
        label: { fontSize: 12, fontWeight: "600" },
