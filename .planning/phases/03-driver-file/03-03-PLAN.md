```typescript
"use client";
import type { TooltipProps } from "recharts";

interface ScorePayload {
  date: string;
  composite: number;
  attendance: number;
  delivery: number;
  financial: number;
  equipment: number;
  platform: number;
}

export default function CustomScoreTooltip({ active, payload }: TooltipProps<number, string>) {
  if (!active || !payload || payload.length === 0) return null;
  const data = payload[0]?.payload as ScorePayload | undefined;
  if (!data) return null;

  return (
    <div className="bg-card border border-sand-300 rounded-xl shadow-lift p-3 text-[12px]">
      <div className="font-medium text-foreground mb-2">{data.date}</div>
      <div className="space-y-1">
        {(["composite", "attendance", "delivery", "financial", "equipment", "platform"] as const).map((k) => (
          <div key={k} className="flex justify-between gap-4">
            <span className="capitalize text-secondary">{k}</span>
            <span className="tabular-nums text-foreground">{data[k] ?? "—"}</span>
          </div>
        ))}
      </div>
    </div>
  );
}
```

**2. frontend/src/components/driver-file/NinetyDayTrendChart.tsx**

```typescript
"use client";
import { useMemo, useState } from "react";
import {
  LineChart,
  Line,
  XAxis,
  YAxis,
  Tooltip,
  CartesianGrid,
  ResponsiveContainer,
  ReferenceDot,
} from "recharts";
import { BarChart3 } from "lucide-react";
import CustomScoreTooltip from "./CustomScoreTooltip";
import type { TrendPoint } from "@/types/driver-file";

const METRICS = [
  { value: "composite", label: "Composite" },
  { value: "attendance", label: "Attendance" },
  { value: "delivery", label: "Delivery" },
  { value: "financial", label: "Financial" },
  { value: "equipment", label: "Equipment" },
  { value: "platform", label: "Platform" },
] as const;

type MetricKey = typeof METRICS[number]["value"];

interface ViolationMarker {
  id: string;
  date: string;
  violationType: string;
}

export default function NinetyDayTrendChart({
  points,
  violations = [],
  hireDate,
}: {
  points: TrendPoint[];
  violations?: ViolationMarker[];
  hireDate?: string | null;
}) {
  const [metric, setMetric] = useState<MetricKey>("composite");
  const [showViolations, setShowViolations] = useState(false);

  const stats = useMemo(() => {
    if (points.length === 0) return null;
    const values = points.map((p) => p[metric]).filter((v): v is number => v != null);
    if (values.length === 0) return null;
    const sum = values.reduce((a, b) => a + b, 0);
    const min = Math.min(...values);
    const max = Math.max(...values);
    const minIdx = points.findIndex((p) => p[metric] === min);
    const maxIdx = points.findIndex((p) => p[metric] === max);
    return {
      avg: Math.round(sum / values.length),
      min,
      minDate: points[minIdx]?.date,
      max,
      maxDate: points[maxIdx]?.date,
    };
  }, [points, metric]);

  // Empty state — Pitfall 1
  if (points.length === 0) {
    return (
      <div className="bg-card border border-sand-200 rounded-2xl shadow-soft p-8 text-center">
        <BarChart3 size={32} className="mx-auto text-sand-400 mb-3" />
        <div className="text-sm font-medium text-foreground mb-1">No 90-day trend yet</div>
        <div className="text-sm text-secondary">First snapshot computes tonight at 23:00 KW.</div>
      </div>
    );
  }

  // <14 days fallback — UI-SPEC §3.4.6
  if (points.length < 14) {
    return (
      <div className="bg-card border border-sand-200 rounded-2xl shadow-soft p-8 text-center">
        <BarChart3 size={32} className="mx-auto text-sand-400 mb-3" />
        <div className="text-sm font-medium text-foreground mb-1">Not enough history yet</div>
        <div className="text-sm text-secondary mt-2">
          Driver has been with you for {points.length} day{points.length === 1 ? "" : "s"}. 90-day trend appears once we have ≥ 14 daily snapshots.
        </div>
        {hireDate && (
          <div className="text-xs text-sand-500 mt-2">Hire date: {hireDate}</div>
        )}
      </div>
    );
  }

  const violationsInWindow = showViolations
    ? violations.filter((v) => points.some((p) => p.date === v.date.slice(0, 10)))
    : [];

  return (
    <div className="bg-card border border-sand-200 rounded-2xl shadow-soft p-6">
      <div className="flex justify-between items-center mb-4 flex-wrap gap-2">
        <div className="text-[11px] uppercase tracking-widest text-secondary font-medium">90-day trend</div>
        <div className="flex items-center gap-3">
          <select
            value={metric}
            onChange={(e) => setMetric(e.target.value as MetricKey)}
            className="rounded-pill h-8 px-3 text-sm border border-sand-200 bg-card hover:bg-sand-50"
            aria-label="Select metric to display"
          >
            {METRICS.map((m) => (
              <option key={m.value} value={m.value}>{m.label}</option>
            ))}
          </select>
          <label className="inline-flex items-center gap-2 text-sm text-secondary">
            <input
              type="checkbox"
              checked={showViolations}
              onChange={(e) => setShowViolations(e.target.checked)}
              className="rounded"
            />
            Show violations
          </label>
        </div>
      </div>
      <div style={{ width: "100%", height: 260 }} dir="ltr">
        <ResponsiveContainer>
          <LineChart data={points} margin={{ top: 8, right: 12, left: 0, bottom: 0 }}>
            <CartesianGrid strokeDasharray="3 3" stroke="#EDEAE5" />
            <XAxis
              dataKey="date"
              stroke="#A99E92"
              fontSize={11}
              tickFormatter={(d: string) => d.slice(5)}
              minTickGap={20}
            />
            <YAxis stroke="#A99E92" fontSize={11} domain={[0, 100]} ticks={[0, 25, 50, 75, 100]} />
            <Tooltip content={<CustomScoreTooltip />} />
            <Line type="monotone" dataKey={metric} stroke="#006838" strokeWidth={2} dot={false} activeDot={{ r: 4 }} connectNulls />
            {violationsInWindow.map((v) => (
              <ReferenceDot
                key={v.id}
                x={v.date.slice(0, 10)}
                y={50}
                r={5}
                fill="#DC2626"
                stroke="#FFFFFF"
                strokeWidth={2}
                ifOverflow="hidden"
              />
            ))}
          </LineChart>
        </ResponsiveContainer>
      </div>
      {stats && (
        <div className="text-xs text-secondary mt-3 flex flex-wrap gap-x-3 gap-y-1">
          <span>Avg: <span className="tabular-nums text-foreground">{stats.avg}</span></span>
          <span>Min: <span className="tabular-nums text-foreground">{stats.min}</span> ({stats.minDate})</span>
          <span>Max: <span className="tabular-nums text-foreground">{stats.max}</span> ({stats.maxDate})</span>
          {violations.length > 0 && (
            <span>{violations.length} violation{violations.length === 1 ? "" : "s"} in 90d</span>
          )}
        </div>
      )}
    </div>
  );
}
```

```typescript
function bucketViolations(items: ViolationItem[]): { open: ViolationItem[]; resolved: ViolationItem[]; appealed: ViolationItem[] } {
  const open: ViolationItem[] = [];
  const resolved: ViolationItem[] = [];
  const appealed: ViolationItem[] = [];
  for (const v of items) {
    if (v.appealStatus === "PENDING" || v.appealStatus === "REJECTED") appealed.push(v);
    else if (v.violationStatus === "OVERTURNED" || v.violationStatus === "EXPIRED") resolved.push(v);
    else open.push(v);
  }
  return { open, resolved, appealed };
}
```

```typescript
import { useMemo, useState } from "react";
import { useSearchParams, useRouter } from "next/navigation";
import { cn } from "@/lib/cn";
import PlatformBadge from "@/components/shared/PlatformBadge";
import StatusBadge from "@/components/shared/StatusBadge";
import type { OrderRow } from "@/types/driver-file";

const PLATFORMS = ["all", "keeta", "talabat", "deliveroo", "americana"] as const;
type PlatformFilter = typeof PLATFORMS[number];

function statusToBadge(s: string) {
  switch (s) {
    case "delivered": return { status: "ON_TIME", label: "delivered" };
    case "cancelled": return { status: "ABSENT", label: "cancelled" };
    case "late": return { status: "LATE", label: "late" };
    case "failed": return { status: "NO_SHOW", label: "failed" };
    default: return { status: s, label: s };
  }
}

export default function RecentOrdersTable({ orders, driverId }: { orders: OrderRow[]; driverId: string }) {
  const router = useRouter();
  const search = useSearchParams();
  const initial = (search.get("platform") as PlatformFilter) ?? "all";
  const [active, setActive] = useState<PlatformFilter>(
    PLATFORMS.includes(initial) ? initial : "all",
  );

  const counts = useMemo(() => {
    const c: Record<PlatformFilter, number> = { all: orders.length, keeta: 0, talabat: 0, deliveroo: 0, americana: 0 };
    for (const o of orders) {
      const k = o.platform.toLowerCase() as PlatformFilter;
      if (k in c) (c as any)[k]++;
    }
    return c;
  }, [orders]);

  const filtered = useMemo(() => {
    if (active === "all") return orders;
    return orders.filter((o) => o.platform.toLowerCase() === active);
  }, [orders, active]);

  const summary = useMemo(() => ({
    delivered: filtered.filter((o) => o.status === "delivered").length,
    cancelled: filtered.filter((o) => o.status === "cancelled").length,
    late: filtered.filter((o) => o.status === "late").length,
  }), [filtered]);

  function changeFilter(next: PlatformFilter) {
    setActive(next);
    const url = new URL(window.location.href);
    if (next === "all") url.searchParams.delete("platform");
    else url.searchParams.set("platform", next);
    router.replace(url.pathname + (url.search || ""));
  }

  return (
    <div className="bg-card border border-sand-200 rounded-2xl shadow-soft p-6">
      <div className="flex justify-between items-center mb-3 flex-wrap gap-2">
        <div className="text-[11px] uppercase tracking-widest text-secondary font-medium">
          Recent orders · last 50
        </div>
        <div className="flex gap-1.5 flex-wrap" role="tablist" aria-label="Filter by platform">
          {PLATFORMS.map((p) => (
            <button
              key={p}
              type="button"
              role="tab"
              aria-selected={active === p}
              onClick={() => changeFilter(p)}
              className={cn(
                "h-7 rounded-pill px-3 text-xs font-medium transition-colors",
                active === p
                  ? "bg-foreground text-white"
                  : "bg-card border border-sand-200 text-sand-700 hover:bg-sand-100",
              )}
            >
              {p === "all" ? "All" : p.charAt(0).toUpperCase() + p.slice(1)}
              <span className="ml-1 tabular-nums opacity-80">{counts[p]}</span>
            </button>
          ))}
        </div>
      </div>
      {filtered.length === 0 ? (
        <div className="text-sm text-secondary text-center py-8">
          {active === "all" ? "No orders yet. Driver may be new or inactive." : `No ${active} orders. Try All.`}
        </div>
      ) : (
        <>
          <table className="w-full text-left">
            <thead className="text-[11px] uppercase tracking-wider text-sand-500 border-b border-sand-200">
              <tr>
                <th className="py-2 px-2 font-medium">Time</th>
                <th className="py-2 px-2 font-medium">Platform</th>
                <th className="py-2 px-2 font-medium">Order ID</th>
                <th className="py-2 px-2 font-medium">Area</th>
                <th className="py-2 px-2 font-medium">Status</th>
              </tr>
            </thead>
            <tbody>
              {filtered.map((o) => {
                const badge = statusToBadge(o.status);
                return (
                  <tr key={o.orderId} className="border-b border-sand-100 last:border-0 hover:bg-sand-50">
                    <td className="py-3 px-2 text-sm tabular-nums text-secondary">
                      {new Date(o.timestamp).toLocaleTimeString("en-US", { hour: "2-digit", minute: "2-digit" })}
                    </td>
                    <td className="py-3 px-2"><PlatformBadge platform={o.platform} /></td>
                    <td className="py-3 px-2 text-sm font-mono text-foreground" title={o.orderId}>
                      {o.orderId.length > 14 ? `${o.orderId.slice(0, 11)}···` : o.orderId}
                    </td>
                    <td className="py-3 px-2 text-sm text-secondary">{o.area ?? "—"}</td>
                    <td className="py-3 px-2"><StatusBadge status={badge.status} label={badge.label} /></td>
                  </tr>
                );
              })}
            </tbody>
          </table>
          <div className="text-sm text-secondary mt-3">
            Showing {filtered.length} most recent · {summary.delivered} delivered · {summary.cancelled} cancelled · {summary.late} late
          </div>
        </>
      )}
    </div>
  );
}
```

```typescript
                <div className="text-sm italic text-foreground leading-relaxed">"{o.text}"</div>
                <div className="text-xs text-secondary mt-2">
                  Observed by Darb · {new Date(o.observedAt).toLocaleDateString("en-US", { dateStyle: "medium" })}
                  {o.confidence != null && ` · confidence ${o.confidence.toFixed(2)}`}
                </div>
              </div>
            ))}
          </div>
        )
      )}

      {tab === "audit" && (
        auditEntries.length === 0 ? (
          <div className="text-sm text-sand-500 text-center py-6">
            No actions taken on this driver yet.
          </div>
        ) : (
          <div>
            {auditEntries.slice(0, 50).map((e) => (
              <button
                key={e.id}
                type="button"
                onClick={() => setSelectedAudit(projectAuditEntry(e))}
                className="w-full text-left border-b border-sand-100 last:border-0 py-3 hover:bg-sand-50 -mx-2 px-2 rounded-md transition-colors"
              >
                <div className="flex items-center justify-between gap-2">
                  <span className="text-sm font-medium text-foreground">{e.toolName}</span>
                  <span className="text-xs text-secondary tabular-nums">
                    {new Date(e.createdAt).toLocaleDateString("en-US", { dateStyle: "short" })}
                  </span>
                </div>
                <div className="text-xs text-secondary mt-1">
                  {e.outcome}{e.approverName && ` · approved by ${e.approverName}`}
                </div>
              </button>
            ))}
            {auditEntries.length > 50 && (
              <a href={`/decisions/audit?driverId=`} className="text-sm font-medium text-primary hover:underline mt-2 block">
                View full audit log →
              </a>
            )}
          </div>
        )
      )}

      <AuditEntryDetail
        entry={selectedAudit}
        open={selectedAudit !== null}
        onClose={() => setSelectedAudit(null)}
        canRollback={false}
      />
    </div>
  );
}
```
