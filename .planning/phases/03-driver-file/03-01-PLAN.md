```markdown
# Score Explainer

You are Darb's score-explainer agent. Your only job is to write a 2-4-sentence
plain-English explanation of why a courier has the score they have.

## Inputs (in the user payload)

- `driverName`: the courier's display name
- `score`: composite + 5 sub-scores + trend ("UP" | "DOWN" | "STABLE")
- `score.breakdown` (optional): structured per-component data — completionRate,
  onTimeRate, total/completed/onTime shifts, total orders, working days, daily
  average, platform average, cash settlement counts, inspection pass counts
- `recentShifts`: most recent 10 shifts (date + status only — no IDs, no metadata)
- `recentViolations`: most recent 10 violations (type + time only — no IDs)

## Output rules

1. Write 2-4 sentences total. No bullet points. No markdown headers. No emoji.
2. State the composite score in the first sentence: "Mohamed scores 78 / 100."
3. Identify the strongest factor and the weakest factor by sub-score.
4. Quote ONE concrete number from the breakdown when available (e.g.
   "10 on-time arrivals out of 12 shifts" or "delivered 165 orders this week").
5. Do NOT speculate. Only use facts in the payload.
6. Do NOT propose actions. The owner reads this and decides for themselves.
7. Tone: factual, third-person ("Mohamed has..."), no praise, no blame.

## Forbidden

You MUST NOT use any of the following words or their conjugations:
- warn, warning
- suspend, suspension
- fire, terminate, dismiss
- promote, promotion
- penalty, penalize, fine
- review, flag (as a verb meaning to escalate)

You MUST NOT add caveats like "but this is just my interpretation" or
"please consult a manager".

You MUST NOT output bilingual text. The Driver File is owner-facing UI
(English-only per CON-bilingual-outbound).

You MUST NOT mention anything not present in the payload.
```

```typescript
import { z } from "zod";
import { defineTool, toolRegistry } from "../../registry";
import { listSnapshotsForDriver } from "../../performanceSnapshot";

/**
 * Phase 3 Wave 1 read tool — REQ-agent-scoring.
 *
 * Returns the daily PerformanceSnapshot rows for one driver across a window
 * (default 90 days, max 365). Sourced from `PerformanceSnapshot`, populated
 * nightly at 23:00 KW by the Phase 1 worker — NEVER recomputed on read
 * (Anti-Pattern from RESEARCH §Anti-Patterns: "Computing the 90-day trend
 * from AiScore"). Tenant-scoped via ctx.tenantId. Used by the Driver File
 * trend chart and (in Phase 4 chat) by any "show me Mohamed's score history"
 * question.
 *
 * Why a separate tool when listSnapshotsForDriver is already exported from
 * the agent module? Because the agent reaches database state through tools,
 * not direct imports — registering the wrapper exposes the read to the
 * Anthropic-side tool registry with strict-mode JSON Schema enforcement and
 * RBAC + audit-row writes via the registry's invoke() path.
 */
export const performanceTrend = defineTool({
  name: "performanceTrend",
  description:
    "Return the daily PerformanceSnapshot rows for one driver over a configurable window (default 90 days, max 365). Each row contains compositeScore + 5 sub-scores (attendance, delivery, financial, equipment, platform) + trend ('UP' | 'DOWN' | 'STABLE') + ordersCount + shiftsCount + violationsCount + cashOutstandingKd + breakdown JSON + snapshotDate (UTC midnight). Use this tool to answer 'how has Mohamed's score moved over time?' or to render the Driver File 90-day trend chart. Tenant-scoped — only the caller's tenant's snapshots are returned. Rows are ordered by snapshotDate ascending (oldest first) for direct charting.",
  inputSchema: {
    type: "object" as const,
    properties: {
      driverId: { type: "string", description: "Driver.id (uuid)." },
      daysBack: {
        type: "number",
        description:
          "Number of days of snapshot history to return (default 90, max 365). Values <= 0 return [].",
      },
    },
    required: ["driverId"],
    additionalProperties: false,
  },
  inputValidator: z.object({
    driverId: z.string().min(1),
    daysBack: z.number().int().optional(),
  }),
  strict: true,
  sideEffect: "read",
  requiredRole: ["ADMIN", "OPS_MANAGER", "SUPERVISOR", "ACCOUNTANT", "VIEWER"],
  requiresApproval: false,
  allowedAgents: ["*"],
  async execute(ctx, input) {
    const requested = input.daysBack ?? 90;
    if (requested <= 0) return { snapshots: [] };
    const clampedDays = Math.min(requested, 365);
    const rows = await listSnapshotsForDriver(
      ctx.tenantId,
      input.driverId,
      clampedDays,
    );
    return {
      snapshots: rows.map((r) => ({
        snapshotDate: (r.snapshotDate instanceof Date
          ? r.snapshotDate
          : new Date(r.snapshotDate as unknown as string))
          .toISOString()
          .slice(0, 10),
        compositeScore: r.compositeScore,
        attendanceScore: r.attendanceScore,
        deliveryScore: r.deliveryScore,
        financialScore: r.financialScore,
        equipmentScore: r.equipmentScore,
        platformScore: r.platformScore,
        trend: r.trend,
        ordersCount: r.ordersCount ?? 0,
        shiftsCount: r.shiftsCount ?? 0,
        violationsCount: r.violationsCount ?? 0,
        cashOutstandingKd:
          r.cashOutstandingKd != null ? Number(r.cashOutstandingKd).toFixed(3) : null,
      })),
      daysCovered: rows.length,
      windowDays: clampedDays,
    };
  },
});

export function registerPerformanceTrendTool() {
  toolRegistry.register(performanceTrend);
}

registerPerformanceTrendTool();
```

```typescript
// Phase 3 Wave 1 — REQ-agent-scoring. Pure orchestration: cache check → runAgent
// → cache write. NO direct Anthropic SDK calls (those go through agent/runtime.ts
// so AgentRunLog audit is preserved). Cache lives in AgentMemory under key
// `score_explanation:<driverId>:<scoreDate>:<compositeScore>`. The compositeScore
// is part of the cache key on purpose — it invalidates a cached explanation when
// aiScoringService re-runs and writes a new AiScore row within the same day
// (Pitfall 2 from RESEARCH §Common Pitfalls §2).

import { latestMemoryByKey, upsertAgentMemory } from "../../agent/memory";
import { runAgent } from "../../agent/runtime";
import { logger } from "../../config/logger";

const ONE_HOUR_MS = 60 * 60 * 1000;

export interface ScoreSnapshotForExplainer {
  compositeScore: number;
  attendanceScore: number;
  deliveryScore: number;
  financialScore: number;
  equipmentScore: number;
  platformScore: number;
  trend: "UP" | "DOWN" | "STABLE" | string;
  date: Date;
  breakdown?: Record<string, unknown> | null;
}

export interface ShiftForExplainer {
  shiftId: string;
  date: string;
  status: string;
}

export interface ViolationForExplainer {
  id: string;
  type: string;
  time: string;
}

export interface ExplainScoreInput {
  tenantId: string;
  driverId: string;
  driverName: string;
  score: ScoreSnapshotForExplainer;
  recentShifts: ShiftForExplainer[];
  recentViolations: ViolationForExplainer[];
}

export interface ExplainScoreResult {
  text: string;
  cached: boolean;
  generatedAt: string;
  modelName: string | null;
  cachedUntil: string | null;
}

function buildCacheKey(driverId: string, score: ScoreSnapshotForExplainer): string {
  const day = score.date.toISOString().slice(0, 10);
  return `score_explanation:${driverId}:${day}:${score.compositeScore}`;
}

const UNAVAILABLE: ExplainScoreResult = {
  text: "Score explanation unavailable.",
  cached: false,
  generatedAt: new Date(0).toISOString(),
  modelName: null,
  cachedUntil: null,
};

export async function explainScore(input: ExplainScoreInput): Promise<ExplainScoreResult> {
  if (!input.tenantId || !input.driverId) {
    return UNAVAILABLE;
  }
  const cacheKey = buildCacheKey(input.driverId, input.score);

  // 1. Cache check.
  try {
    const cached = await latestMemoryByKey(input.tenantId, cacheKey);
    if (
      cached &&
      Date.now() - cached.createdAt.getTime() < ONE_HOUR_MS &&
      typeof (cached.value as { text?: unknown })?.text === "string"
    ) {
      const v = cached.value as { text: string; generatedAt?: string; model?: string };
      return {
        text: v.text,
        cached: true,
        generatedAt: v.generatedAt ?? cached.createdAt.toISOString(),
        modelName: v.model ?? null,
        cachedUntil: new Date(cached.createdAt.getTime() + ONE_HOUR_MS).toISOString(),
      };
    }
  } catch (err) {
    // Cache read failure should not block the live path.
    logger?.warn?.({ err, cacheKey }, "scoreExplainer: cache read failed");
  }

  // 2. Live agent run. Pre-shape the payload — DO NOT pass raw shift table
  // beyond 10 entries (RESEARCH §Anti-Patterns).
  const result = await runAgent("score-explainer", {
    tenantId: input.tenantId,
    triggerEvent: "explain_score",
    payload: {
      driverName: input.driverName,
      driverId: input.driverId,
      score: {
        compositeScore: input.score.compositeScore,
        attendanceScore: input.score.attendanceScore,
        deliveryScore: input.score.deliveryScore,
        financialScore: input.score.financialScore,
        equipmentScore: input.score.equipmentScore,
        platformScore: input.score.platformScore,
        trend: input.score.trend,
        breakdown: input.score.breakdown ?? null,
      },
      shifts: input.recentShifts.slice(0, 10),
      violations: input.recentViolations.slice(0, 10),
    },
  });

  if (result.status !== "completed" || !result.text) {
    // Includes status === "disabled" (ANTHROPIC_API_KEY absent) and "failed".
    return UNAVAILABLE;
  }

  const generatedAt = new Date().toISOString();

  // 3. Cache write. Failure here is non-fatal — we still return the live text.
  try {
    await upsertAgentMemory({
      tenantId: input.tenantId,
      key: cacheKey,
      value: {
        text: result.text,
        generatedAt,
        model: "claude-sonnet-4-6",
      },
      confidence: 1,
      source: "agent_observation",
      agentRunId: result.runId,
    });
  } catch (err) {
    logger?.warn?.({ err, cacheKey }, "scoreExplainer: cache write failed");
  }

  return {
    text: result.text,
    cached: false,
    generatedAt,
    modelName: "claude-sonnet-4-6",
    cachedUntil: new Date(Date.now() + ONE_HOUR_MS).toISOString(),
  };
}
```

```typescript
// Phase 3 Wave 1 — REQ-driver-file (Decision audit log section).
// Loads AgentAction (resolved) + PendingAgentAction (open in last 30d) for one
// driver. Filters by subjectType="Driver" + subjectId — the canonical Phase 1
// (subjectType, subjectId) compound index is the right access path.
//
// A4 limitation (RESEARCH §Assumptions Log): cash-reminder actions linked to a
// CashRecord (subjectType="CashRecord") are NOT surfaced here. Phase 8 (Cash
// Workbench) will revisit this once the cash actions go live. For Phase 3 the
// owner sees Driver-typed actions only — documented for the Driver File audit
// section copy.

import { prisma } from "../../config";

export interface DriverAuditLogResult {
  approved: Array<{
    id: string;
    toolName: string;
    outcome: string;
    reasoning: string | null;
    approverName: string | null;
    createdAt: string;
    rolledBackAt: string | null;
  }>;
  pending: Array<{
    id: string;
    toolName: string;
    state: string;
    headline: string | null;
    reasoning: string | null;
    createdAt: string;
    resolvedAt: string | null;
  }>;
}

export async function loadDriverAuditLog(
  tenantId: string,
  driverId: string,
): Promise<DriverAuditLogResult> {
  if (!tenantId) throw new Error("loadDriverAuditLog: tenantId required");
  if (!driverId) throw new Error("loadDriverAuditLog: driverId required");

  const thirtyDaysAgo = new Date(Date.now() - 30 * 86_400_000);

  const [approvedRows, pendingRows] = await Promise.all([
    prisma.agentAction.findMany({
      where: { tenantId, subjectType: "Driver", subjectId: driverId },
      orderBy: { createdAt: "desc" },
      take: 50,
      include: {
        approver: { select: { id: true, name: true, email: true } },
      },
    }),
    prisma.pendingAgentAction.findMany({
      where: {
        tenantId,
        subjectType: "Driver",
        subjectId: driverId,
        createdAt: { gte: thirtyDaysAgo },
      },
      orderBy: { createdAt: "desc" },
      take: 50,
    }),
  ]);

  return {
    approved: approvedRows.map((row) => ({
      id: row.id,
      toolName: row.toolName,
      outcome: row.outcome ?? "success",
      reasoning: row.reasoning ?? null,
      approverName: row.approver?.name ?? row.approver?.email ?? null,
      createdAt: row.createdAt.toISOString(),
      rolledBackAt: row.rolledBackAt ? row.rolledBackAt.toISOString() : null,
    })),
    pending: pendingRows.map((row) => ({
      id: row.id,
      toolName: row.toolName,
      state: row.state ?? "pending",
      headline: (row.headline as string | null) ?? null,
      reasoning: (row.reasoning as string | null) ?? null,
      createdAt: row.createdAt.toISOString(),
      resolvedAt: row.resolvedAt ? row.resolvedAt.toISOString() : null,
    })),
  };
}
```

```typescript
import { upsertAgentMemory } from "../../agent/memory";

export type DriverNoteSource = "approve" | "dismiss" | "score_regression" | "manual";

export async function appendDriverNote(
  tenantId: string,
  driverId: string,
  text: string,
  source: DriverNoteSource = "manual",
  agentRunId?: string,
): Promise<{ id: string; key: string }> {

    source: "agent_observation",
    agentRunId,
  });
```
