**2. frontend/src/components/shared/DriverLink.tsx — global linkability primitive**

Copy UI-SPEC §2.2 verbatim — there is a complete component spec there. Adapt the prop name `driverName` to `name` if (and only if) the Wave 0 RED test asserts `name`; otherwise keep `driverName`. (Read DriverLink.test.tsx and match.)

```typescript
"use client";
import Link from "next/link";
import { cn } from "@/lib/cn";

interface DriverLinkProps {
  driverId: string;
  name: string;            // Wave 0 test asserts `name` prop
  platform?: "keeta" | "talabat" | "deliveroo" | "americana";
  variant?: "inline" | "card-headline" | "table-cell" | "subtle";
  showPlatformDot?: boolean;
  className?: string;
}

export default function DriverLink({
  driverId,
  name,
  platform,
  variant = "inline",
  showPlatformDot = false,
  className,
}: DriverLinkProps) {
  const href = platform
    ? `/drivers/${driverId}?from=${platform}`
    : `/drivers/${driverId}`;
  const baseClass = cn(
    "inline-flex items-center gap-1.5 transition-colors",
    variant === "inline" && "text-primary hover:underline decoration-primary/40 underline-offset-2 font-medium",
    variant === "card-headline" && "text-foreground font-semibold hover:text-primary",
    variant === "table-cell" && "text-foreground hover:text-primary",
    variant === "subtle" && "text-secondary hover:text-foreground",
    className,
  );
  return (
    <Link href={href} className={baseClass} prefetch={false}>
      {showPlatformDot && platform && (
        <span
          className={cn(
            "w-1.5 h-1.5 rounded-full",
            platform === "keeta" && "bg-keeta",
            platform === "talabat" && "bg-talabat",
            platform === "deliveroo" && "bg-deliveroo",
            platform === "americana" && "bg-americana",
          )}
        />
      )}
      {name}
    </Link>
  );
}
```

```typescript
"use client";
import { useEffect, useState } from "react";
import { cn } from "@/lib/cn";

interface SectionSpec { id: string; label: string; }

export default function DriverFileScrollSpy({ sections }: { sections: SectionSpec[] }) {
  const [activeId, setActiveId] = useState(sections[0]?.id ?? "");

  useEffect(() => {
    const observer = new IntersectionObserver(
      (entries) => {
        const visible = entries.filter((e) => e.isIntersecting);
        if (visible.length > 0) {
          // Pick the section closest to the top of the viewport
          visible.sort((a, b) => a.boundingClientRect.top - b.boundingClientRect.top);
          setActiveId(visible[0].target.id);
        }
      },
      { rootMargin: "-25% 0px -50% 0px" }
    );
    sections.forEach((s) => {
      const el = document.getElementById(s.id);
      if (el) observer.observe(el);
    });
    return () => observer.disconnect();
  }, [sections]);

  return (
    <nav className="space-y-1 text-sm" aria-label="Section navigation">
      {sections.map((s) => (
        <a
          key={s.id}
          href={`#${s.id}`}
          aria-current={activeId === s.id ? "location" : undefined}
          className={cn(
            "block py-1 transition-colors focus:ring-2 focus:ring-primary/40 rounded-pill",
            activeId === s.id
              ? "text-primary font-medium"
              : "text-secondary hover:text-foreground"
          )}
        >
          {activeId === s.id ? "● " : "○ "}
          {s.label}
        </a>
      ))}
    </nav>
  );
}
```

```typescript
"use client";
import { TrendingUp, TrendingDown, Minus } from "lucide-react";

function bandColor(score: number): string {
  if (score < 40) return "#DC2626"; // red-600
  if (score < 70) return "#F59E0B"; // amber-500
  return "#006838";                  // primary forest green
}

export default function ScoreGauge({ value, delta = 0 }: { value: number; delta?: number }) {
  // 200deg sweep from -100deg to +100deg. Convert to circumference percentage.
  const SWEEP_DEG = 200;
  const RADIUS = 70;
  const STROKE = 12;
  const CIRCUMFERENCE = 2 * Math.PI * RADIUS;
  const ARC_LENGTH = (CIRCUMFERENCE * SWEEP_DEG) / 360;
  const FILL_LENGTH = (ARC_LENGTH * Math.max(0, Math.min(100, value))) / 100;
  const fillColor = bandColor(value);

  const TrendIcon = delta > 0 ? TrendingUp : delta < 0 ? TrendingDown : Minus;
  const deltaColor = delta > 0 ? "text-primary" : delta < 0 ? "text-red-600" : "text-sand-700";

  return (
    <div className="flex flex-col items-center">
      <div
        className="relative"
        role="meter"
        aria-valuenow={value}
        aria-valuemin={0}
        aria-valuemax={100}
        aria-label="Composite performance score"
      >
        <svg width={160} height={120} viewBox="-90 -90 180 180">
          {/* Track */}
          <circle
            r={RADIUS}
            fill="none"
            stroke="#EDEAE5"
            strokeWidth={STROKE}
            strokeLinecap="round"
            strokeDasharray={`${ARC_LENGTH} ${CIRCUMFERENCE}`}
            transform="rotate(-190)"
          />
          {/* Fill */}
          <circle
            r={RADIUS}
            fill="none"
            stroke={fillColor}
            strokeWidth={STROKE}
            strokeLinecap="round"
            strokeDasharray={`${FILL_LENGTH} ${CIRCUMFERENCE}`}
            transform="rotate(-190)"
          />
        </svg>
        <div className="absolute inset-0 flex flex-col items-center justify-center">
          <div className="font-display text-3xl text-foreground tabular-nums">{value}</div>
          <div className="text-xs text-secondary">/ 100</div>
        </div>
      </div>
      {delta !== 0 && (
        <div className={`text-xs font-medium mt-2 inline-flex items-center gap-1 ${deltaColor}`}>
          <TrendIcon size={12} />
          {delta > 0 ? `+${delta}` : delta} vs 7d
        </div>
      )}
    </div>
  );
}
```
