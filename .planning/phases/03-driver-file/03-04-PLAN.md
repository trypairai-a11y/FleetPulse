**2. backend/src/routes/drivers.ts — add GET /:id/score-explanation handler**

After the `/:id/file` handler (Wave 1) and before the existing `/:id` handler, insert:

```typescript
/**
 * GET /api/drivers/:id/score-explanation?refresh=0|1
 *
 * Phase 3 Wave 4 — standalone endpoint for the AskDarbWhyDrawer Refresh
 * button. Reuses Wave 1's explainScore service. ?refresh=1 forces a cache
 * miss; ?refresh=0 (default) honors the 1h AgentMemory cache.
 *
 * Tenant-scoped via the existing router-level middleware. 404 on cross-tenant
 * — same Pitfall 4 mitigation as /file.
 */
router.get("/:id/score-explanation", async (req: Request, res: Response) => {
  try {
    const tenantId = req.user!.tenantId;
    const { id } = req.params;
    const refresh = req.query.refresh === "1";

    const driver = await prisma.driver.findFirst({
      where: { id, tenantId },
      select: { id: true, name: true },
    });
    if (!driver) return res.status(404).json({ error: "Driver not found" });

    const latestScore = await prisma.aiScore.findFirst({
      where: { tenantId, driverId: id },
      orderBy: { date: "desc" },
    });
    if (!latestScore) {
      return res.json({
        text: "Score not yet available — the daily snapshot worker has not yet run for this driver.",
        cached: false,
        generatedAt: null,
        modelName: null,
        cachedUntil: null,
      });
    }

    const [recentShifts, recentViolations] = await Promise.all([
      prisma.shift.findMany({
        where: { tenantId, driverId: id },
        orderBy: { date: "desc" },
        take: 10,
        select: { id: true, date: true, status: true },
      }),
      prisma.violation.findMany({
        where: { tenantId, driverId: id },
        orderBy: { violationTime: "desc" },
        take: 10,
        select: { id: true, violationType: true, violationTime: true },
      }),
    ]);

    const explained = await explainScore({
      tenantId,
      driverId: id,
      driverName: driver.name,
      score: {
        compositeScore: latestScore.compositeScore,
        attendanceScore: latestScore.attendanceScore,
        deliveryScore: latestScore.deliveryScore,
        financialScore: latestScore.financialScore,
        equipmentScore: latestScore.equipmentScore,
        platformScore: latestScore.platformScore,
        trend: latestScore.trend ?? "STABLE",
        date: latestScore.date,
        breakdown: latestScore.breakdown as Record<string, unknown> | null,
      },
      recentShifts: recentShifts.map((s) => ({
        shiftId: s.id,
        date: s.date.toISOString().slice(0, 10),
        status: s.status,
      })),
      recentViolations: recentViolations.map((v) => ({
        id: v.id,
        type: v.violationType,
        time: v.violationTime.toISOString(),
      })),
      forceRefresh: refresh,
    });

    return res.json(explained);
  } catch (err: unknown) {
    const message = (err as Error)?.message ?? "Internal error";
    console.error("[driverFile] /:id/score-explanation failed:", message);
    res.status(500).json({ error: message });
  }
});
```

**3. frontend/src/components/driver-file/AskDarbWhyDrawer.tsx**

```typescript
"use client";
import { useState } from "react";
import { Sparkles, RefreshCw, X } from "lucide-react";
import { useApiQuery } from "@/hooks/useApi";
import type { ScoreExplanation } from "@/types/driver-file";

interface AskDarbWhyDrawerProps {
  driverId: string;
  preWarmed: ScoreExplanation;       // From the bulk /file response
}

export default function AskDarbWhyDrawer({ driverId, preWarmed }: AskDarbWhyDrawerProps) {
  const [open, setOpen] = useState(false);
  const [refreshTick, setRefreshTick] = useState(0);

  // Only fetch once the user opens the drawer AND wants a refresh.
  // The first open uses the pre-warmed text from the bulk endpoint.
  const wantFresh = open && refreshTick > 0;
  const { data: live, isLoading: isFetching } = useApiQuery<ScoreExplanation>(
    ["driver-score-explanation", driverId, refreshTick],
    `/api/drivers/${driverId}/score-explanation${refreshTick > 0 ? "?refresh=1" : ""}`,
    {
      enabled: wantFresh,
      staleTime: 60 * 60 * 1000,        // matches backend 1h cache
      refetchOnWindowFocus: false,
    },
  );

  const explanation: ScoreExplanation = live ?? preWarmed;

  return (
    <div>
      <button
        type="button"
        onClick={() => setOpen((v) => !v)}
        className="inline-flex items-center gap-1 text-sm font-medium text-primary hover:underline focus:ring-2 focus:ring-primary/40 rounded-pill px-1"
        aria-expanded={open}
      >
        <Sparkles size={12} />
        Ask Darb why this score?
      </button>
      {open && (
        <div className="bg-primary/5 border border-primary/15 rounded-xl p-4 mt-3 relative">
          <div className="flex items-center justify-between mb-2">
            <div className="text-[11px] uppercase tracking-widest text-primary font-medium">
              Darb's read on this score
            </div>
            <button
              type="button"
              onClick={() => setOpen(false)}
              className="text-secondary hover:text-foreground"
              aria-label="Close drawer"
            >
              <X size={14} />
            </button>
          </div>
          {isFetching ? (
            <div className="text-sm text-primary animate-pulse">Darb is thinking…</div>
          ) : explanation.text === "Score explanation unavailable." ? (
            <div className="text-sm text-red-700">
              Couldn't generate explanation — try again in a moment.
            </div>
          ) : (
            <div className="text-sm text-foreground leading-relaxed whitespace-pre-line">
              {explanation.text}
            </div>
          )}
          <div className="text-[10px] font-mono text-sand-500 mt-3 flex items-center justify-between">
            <span>
              {explanation.generatedAt && (
                <>Generated {new Date(explanation.generatedAt).toLocaleTimeString("en-US", { hour: "2-digit", minute: "2-digit" })} </>
              )}
              {explanation.cached ? "· cached" : "· fresh"}
              {explanation.modelName && ` · ${explanation.modelName}`}
            </span>
            <button
              type="button"
              onClick={() => setRefreshTick((t) => t + 1)}
              disabled={isFetching}
              className="inline-flex items-center gap-1 text-primary hover:underline disabled:opacity-50 disabled:cursor-not-allowed"
              aria-label="Refresh explanation"
            >
              <RefreshCw size={10} />
              Refresh
            </button>
          </div>
        </div>
      )}
    </div>
  );
}
```

If `ScoreExplanation` interface lacks `modelName`, add it in `types/driver-file.ts` (Wave 1 service returns it; Wave 2 type definition may have stubbed it null).

// backend/src/__tests__/integration/driverFileSmoke.test.ts
//
// Phase 3 Wave 4 — end-to-end smoke against the design-partner-1 fixture.
// Gated on FIXTURE_SEEDED=true so it doesn't fail on unseeded local DBs.
//
// Run with:
//   FIXTURE_SEEDED=true npm test -- --testPathPatterns=driverFileSmoke
//
// Pre-requisite: `cd backend && npm run seed:design-partner-fixture` (Phase 2).

import request from "supertest";
import { prisma } from "../../config";
import app from "../../server";       // or whatever exports the Express app — check the existing routes test
import { signTestJwt } from "../helpers/auth";  // Phase 2 helper; if absent, copy the pattern from driversFile.test.ts

const FIXTURE_SEEDED = process.env.FIXTURE_SEEDED === "true";

(FIXTURE_SEEDED ? describe : describe.skip)("Driver File smoke against design-partner-1 fixture", () => {
  let drivers: Array<{ id: string; tenantId: string; platform: string }> = [];
  let authToken: string | null = null;
  let tenantId: string | null = null;

  beforeAll(async () => {
    // Locate the design-partner-1 fixture tenant + super-admin user.
    const tenant = await prisma.tenant.findFirst({
      where: { slug: { startsWith: "design-partner" } },
      select: { id: true },
    });
    if (!tenant) {
      console.warn("[driverFileSmoke] no design-partner tenant found — fixture not seeded; skipping");
      return;
    }
    tenantId = tenant.id;

    drivers = await prisma.driver.findMany({
      where: { tenantId: tenant.id },
      take: 10,
      select: { id: true, tenantId: true, platform: true },
    });
    if (drivers.length === 0) {
      console.warn("[driverFileSmoke] no drivers in fixture — skipping");
      return;
    }

    // Borrow the auth shape used by driversFile.test.ts. If the helper does
    // not exist, mint a JWT inline using config.JWT_SECRET.
    authToken = signTestJwt({ tenantId: tenant.id, role: "ADMIN", userId: "smoke-runner" });
  });

  it("returns 200 + 10 keys for every seeded driver", async () => {
    if (!authToken || drivers.length === 0) return;

    const start = Date.now();
    for (const d of drivers) {
      const res = await request(app)
        .get(`/api/drivers/${d.id}/file`)
        .set("Authorization", `Bearer ${authToken}`);

      expect(res.status).toBe(200);

      const expectedKeys = [
        "profile",
        "liveStatus",
        "score",
        "scoreExplanation",
        "snapshots90d",
        "attendance",
        "cash",
        "violations",
        "agentNotes",
        "decisionAuditLog",
      ];
      for (const k of expectedKeys) {
        expect(res.body).toHaveProperty(k);
      }
      expect(res.body.profile.id).toBe(d.id);
      expect(Array.isArray(res.body.snapshots90d)).toBe(true);
      expect(Array.isArray(res.body.agentNotes)).toBe(true);
      expect(res.body.decisionAuditLog).toHaveProperty("approved");
      expect(res.body.decisionAuditLog).toHaveProperty("pending");
      expect(Array.isArray(res.body.decisionAuditLog.approved)).toBe(true);
      expect(Array.isArray(res.body.decisionAuditLog.pending)).toBe(true);
    }
    const elapsed = Date.now() - start;
    expect(elapsed).toBeLessThan(5000);
    console.info(`[driverFileSmoke] 8 drivers in ${elapsed}ms — avg ${Math.round(elapsed / drivers.length)}ms/driver`);
  });

  it("cross-tenant request returns 404", async () => {
    if (!authToken || drivers.length === 0) return;

    // Find a driver from a DIFFERENT tenant and confirm 404.
    const otherDriver = await prisma.driver.findFirst({
      where: { tenantId: { not: tenantId! } },
      select: { id: true },
    });
    if (!otherDriver) {
      console.warn("[driverFileSmoke] no cross-tenant driver to test — skipping");
      return;
    }
    const res = await request(app)
      .get(`/api/drivers/${otherDriver.id}/file`)
      .set("Authorization", `Bearer ${authToken}`);
    expect(res.status).toBe(404);
    expect(res.body).toEqual({ error: "Driver not found" });
  });
});
