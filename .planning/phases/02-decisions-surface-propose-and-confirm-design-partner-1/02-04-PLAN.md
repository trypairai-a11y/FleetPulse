import { prisma } from "../../config";

const PRICE_PER_COURIER_KD = 2.0;
const FLOOR_KD = 200.0;
const ACTIVE_HOURS_THRESHOLD_MINUTES = 240;

export interface MonthlyBill {
  tenantId: string;
  month: string;
  monthlyActiveCouriers: number;
  computedKd: number;
  override: number | null;
  netKd: number;
  designPartner: boolean;
  trialEndsAt: Date | null;
}

export async function isCourierActiveThisMonth(
  tenantId: string,
  driverId: string,
  monthDate: Date
): Promise<boolean> {
  const start = new Date(Date.UTC(monthDate.getUTCFullYear(), monthDate.getUTCMonth(), 1));
  const end = new Date(Date.UTC(monthDate.getUTCFullYear(), monthDate.getUTCMonth() + 1, 1));
  const row = await prisma.shift.findFirst({
    where: {
      tenantId,
      driverId,
      date: { gte: start, lt: end },
      actualHoursMinutes: { gte: ACTIVE_HOURS_THRESHOLD_MINUTES },
    },
    select: { id: true },
  });
  return row !== null;
}

export async function computeMonthlyBill(
  tenantId: string,
  monthDate: Date
): Promise<MonthlyBill> {
  const start = new Date(Date.UTC(monthDate.getUTCFullYear(), monthDate.getUTCMonth(), 1));
  const end = new Date(Date.UTC(monthDate.getUTCFullYear(), monthDate.getUTCMonth() + 1, 1));

  const groups = await prisma.shift.groupBy({
    by: ["driverId"],
    where: {
      tenantId,
      date: { gte: start, lt: end },
      actualHoursMinutes: { gte: ACTIVE_HOURS_THRESHOLD_MINUTES },
    },
    _count: { id: true },
  });
  const monthlyActiveCouriers = groups.length;

  const tenant = await prisma.tenant.findUnique({
    where: { id: tenantId },
    select: { designPartner: true, monthlyOverrideKd: true, trialEndsAt: true, name: true },
  });
  if (!tenant) throw new Error(`Tenant not found: ${tenantId}`);

  const computedKd = Math.max(monthlyActiveCouriers * PRICE_PER_COURIER_KD, FLOOR_KD);
  const override = tenant.monthlyOverrideKd ? Number(tenant.monthlyOverrideKd) : null;
  const netKd = override ?? computedKd;

  return {
    tenantId,
    month: `${monthDate.getUTCFullYear()}-${String(monthDate.getUTCMonth() + 1).padStart(2, "0")}`,
    monthlyActiveCouriers,
    computedKd,
    override,
    netKd,
    designPartner: tenant.designPartner ?? false,
    trialEndsAt: tenant.trialEndsAt,
  };
}

export async function listMonthlyBillsAcrossTenants(monthDate: Date): Promise<MonthlyBill[]> {
  const tenants = await prisma.tenant.findMany({ select: { id: true } });
  return Promise.all(tenants.map((t) => computeMonthlyBill(t.id, monthDate)));
}

import { Worker, Queue } from "bullmq";
import IORedis from "ioredis";
import { prisma } from "../config";
import { env } from "../config/env";
import { logger } from "../config/logger";
import { buildOnboardingReport } from "../services/onboarding/onboardingReport";

export const ONBOARDING_BACKWASH_QUEUE = "onboarding-backwash";

interface BackwashJobData {
  tenantId: string;
  daysBack: number;
  platforms: ("KEETA" | "TALABAT" | "DELIVEROO" | "AMERICANA")[];
}

interface BackwashJobResult {
  reportData: unknown;  // ReportData; serialised through Redis
  summary: { ordersIngested: number; couriers: number; durationMs: number };
}

const CHUNK_DAYS = 5;
const CONCURRENCY = 2;

export function getOnboardingBackwashQueue(): Queue<BackwashJobData> | null {
  if (!env.REDIS_URL || env.REDIS_URL === "redis://localhost:6379") {
    logger.warn("ONBOARDING_BACKWASH_QUEUE: REDIS_URL not configured; queue disabled");
    return null;
  }
  const connection = new IORedis(env.REDIS_URL, { maxRetriesPerRequest: null });
  return new Queue<BackwashJobData>(ONBOARDING_BACKWASH_QUEUE, { connection });
}

export function startOnboardingBackwashWorker(): Worker<BackwashJobData, BackwashJobResult> | null {
  if (!env.REDIS_URL || env.REDIS_URL === "redis://localhost:6379") {
    logger.warn("startOnboardingBackwashWorker: REDIS_URL not configured; worker disabled");
    return null;
  }
  const connection = new IORedis(env.REDIS_URL, { maxRetriesPerRequest: null });

  return new Worker<BackwashJobData, BackwashJobResult>(
    ONBOARDING_BACKWASH_QUEUE,
    async (job) => {
      const { tenantId, daysBack, platforms } = job.data;
      const startTime = Date.now();
      const totalSteps = platforms.length + 1; // platforms + final report build
      let stepsDone = 0;
      let totalOrdersIngested = 0;

      for (const platform of platforms) {
        await job.updateProgress({
          step: stepsDone,
          totalSteps,
          message: `Pulling ${daysBack} days of ${platform} data`,
        });
        const ordersIngested = await runBackwashForPlatform(tenantId, platform, daysBack);
        totalOrdersIngested += ordersIngested;
        stepsDone++;
      }

      await job.updateProgress({
        step: stepsDone,
        totalSteps,
        message: "Building Darb's read on your fleet",
      });

      const reportData = await buildOnboardingReport(tenantId, daysBack);
      stepsDone++;

      const courierCount = await prisma.driver.count({ where: { tenantId, isActive: true } });
      return {
        reportData,

async function runBackwashForPlatform(
  tenantId: string,
  platform: "KEETA" | "TALABAT" | "DELIVEROO" | "AMERICANA",
  daysBack: number
): Promise<number> {
  // Chunk the day window into 5-day pieces.
  const now = new Date();
  const windowStart = new Date(now);
  windowStart.setUTCDate(windowStart.getUTCDate() - daysBack);

  let totalOrders = 0;
  for (let dayOffset = 0; dayOffset < daysBack; dayOffset += CHUNK_DAYS) {
    const chunkStart = new Date(windowStart);
    chunkStart.setUTCDate(chunkStart.getUTCDate() + dayOffset);
    const chunkEnd = new Date(chunkStart);
    chunkEnd.setUTCDate(chunkEnd.getUTCDate() + CHUNK_DAYS);

    const chunkOrders = await ingestChunkForPlatform(tenantId, platform, chunkStart, chunkEnd);
    totalOrders += chunkOrders;
  }
  return totalOrders;
}

async function ingestChunkForPlatform(
  tenantId: string,

  // that confirms data is present.
  const count = await prisma.orderLog.count({
    where: {
      tenantId,
      platform,
      createdAt: { gte: chunkStart, lt: chunkEnd },
    },
  });
  return count;
}
