    ```typescript
    import { Router } from "express";
    import { authMiddleware } from "../middleware/auth";
    import { tenantScope } from "../middleware/tenantScope";
    import { runAgent } from "../agent";

    export type CosMode = "morning_brief" | "anomalies" | "actions" | "ask";

    const router = Router();
    router.use(authMiddleware, tenantScope);

    // POST /api/ai/cos — generic prompt entry point. Maps the incoming `mode`
    // to a registered agent: morning_brief/anomalies/actions go to "narrator",
    // ask goes to "chat".
    router.post("/", async (req, res) => {
      try {
        const tenantId = req.user!.tenantId;
        const mode = (req.body?.mode ?? "ask") as CosMode;
        const message = (req.body?.message ?? "") as string;
        const agentId = mode === "ask" ? "chat" : "narrator";
        const result = await runAgent(agentId, {
          tenantId,
          triggerEvent: `route:cos:${mode}`,
          userMessage: message || `Run ${mode} for tenant ${tenantId}`,
        });
        return res.json({
          status: result.status,
          text: result.text,
          runId: result.runId,
          actionsProposed: result.actionsProposed,
          pendingActionIds: result.pendingActionIds,
        });
      } catch (err: any) {
        return res.status(500).json({ error: err?.message ?? "agent_run_failed" });
      }
    });

    // POST /api/ai/cos/daily — convenience for the daily briefing.
    router.post("/daily", async (req, res) => {
      try {
        const tenantId = req.user!.tenantId;
        const result = await runAgent("narrator", {
          tenantId,
          triggerEvent: "route:cos:daily",
          payload: { kind: "daily_briefing" },
        });
        return res.json({
          status: result.status,
          text: result.text,
          runId: result.runId,
        });
      } catch (err: any) {
        return res.status(500).json({ error: err?.message ?? "agent_run_failed" });
      }
    });

    export default router;
    ```
